    24  func wordWrapLine(columns int, line string) string {
    25  	var prefix, suffix string
    26  	firstNonSpace := strings.IndexFunc(line, nonSpace)
    27  	lastNonSpace := strings.LastIndexFunc(line, nonSpace)
    28  	switch {
    29  	case firstNonSpace == -1 && lastNonSpace == -1:
    30  		return line // only space chars found, return immediately
    31  	case firstNonSpace != -1 && lastNonSpace != -1:
    32  		prefix, suffix = line[:firstNonSpace], line[lastNonSpace+1:]
    33  		line = line[firstNonSpace : lastNonSpace+1]
    34  	default:
    35  		panic("Impossible case. Either no non-space runes are found or none are found.")
    36  	}
    37  
    38  	lines := wordWrap(columns, line)
    39  	return prefix + strings.Join(lines, "\n"+prefix) + suffix

    43  	s = strings.TrimSpace(s)
