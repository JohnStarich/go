    18  	watcher, err := fsnotify.NewWatcher()
    19  	if err != nil {
    20  		return err

    23  	// increase soft open file limit to maximum
    24  	var rLimit syscall.Rlimit
    25  	if err := syscall.Getrlimit(syscall.RLIMIT_NOFILE, &rLimit); err != nil {
    26  		return errors.Wrap(err, "Failed to get open file limit")
    27  	}
    28  	if rLimit.Cur < rLimit.Max && rLimit.Cur < maxOpenFiles {
    29  		rLimit.Cur = min(rLimit.Max, maxOpenFiles)
    30  		err := syscall.Setrlimit(syscall.RLIMIT_NOFILE, &rLimit)
    31  		if err != nil {
    32  			return errors.Wrap(err, "Failed to increase soft open file limit")
    33  		}
    34  	}
    35  
    36  	err = filepath.Walk(path, func(path string, info os.FileInfo, err error) error {
    37  		if err != nil {
    38  			return err
    39  		}
    40  		if info.IsDir() {
    41  			return watcher.Add(path)
    42  		}
    43  		return nil
    44  	})
    45  	if err != nil {
    46  		watcher.Close()
    47  		return err
    48  	}
    49  
    50  	go func() {
    51  		timer := time.NewTimer(0) // fire watch right away
    52  		defer timer.Stop()
    53  
    54  		const debounce = 2 * time.Second
    55  		for {
    56  			select {
    57  			case <-timer.C:
    58  				log.Println("Running watch call...")
    59  				err := do()
    60  				if err != nil {
    61  					log.Println("Error running watch call:", err)
    62  				}
    63  			case <-ctx.Done():
    64  				watcher.Close()
    65  				return
    66  			case event := <-watcher.Events:
    67  				switch {
    68  				case event.Op&fsnotify.Write == fsnotify.Write,
    69  					event.Op&fsnotify.Create == fsnotify.Create:
    70  					timer.Reset(debounce)
    71  				}
    72  			case err := <-watcher.Errors:
    73  				log.Println("error:", err)
    74  			}
    75  		}
    76  	}()
