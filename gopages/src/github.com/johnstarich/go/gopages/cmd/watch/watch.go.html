    18  const maxOpenFiles = 1 << 14 // 2^14 appears to be the maximum value for macOS and 2^20 on linux

    20  func watch(ctx context.Context, path string, do func() error) error {
    21  	watcher, err := fsnotify.NewWatcher()
    22  	if err != nil {
    23  		return err
    24  	}
    25  
    26  	// increase soft open file limit to maximum
    27  	var rLimit syscall.Rlimit
    28  	if err := syscall.Getrlimit(syscall.RLIMIT_NOFILE, &rLimit); err != nil {
    29  		return errors.Wrap(err, "Failed to get open file limit")
    30  	}
    31  	if rLimit.Cur < rLimit.Max && rLimit.Cur < maxOpenFiles {
    32  		rLimit.Cur = min(rLimit.Max, maxOpenFiles)
    33  		err := syscall.Setrlimit(syscall.RLIMIT_NOFILE, &rLimit)
    34  		if err != nil {
    35  			return errors.Wrap(err, "Failed to increase soft open file limit")
    36  		}
    37  	}
    38  
    39  	err = filepath.Walk(path, func(path string, info os.FileInfo, err error) error {
    40  		if err != nil {
    41  			return err
    42  		}
    43  		if info.IsDir() {
    44  			return watcher.Add(path)
    45  		}
    46  		return nil
    47  	})
    48  	if err != nil {
    49  		watcher.Close()
    50  		return err
    51  	}
    52  
    53  	go func() {
    54  		timer := time.NewTimer(0) // fire watch right away
    55  		defer timer.Stop()
    56  
    57  		const debounce = 2 * time.Second
    58  		for {
    59  			select {
    60  			case <-timer.C:
    61  				log.Println("Running watch call...")
    62  				err := do()
    63  				if err != nil {
    64  					log.Println("Error running watch call:", err)
    65  				}
    66  			case <-ctx.Done():
    67  				watcher.Close()
    68  				return
    69  			case event := <-watcher.Events:
    70  				switch {
    71  				case event.Op&fsnotify.Write == fsnotify.Write,
    72  					event.Op&fsnotify.Create == fsnotify.Create:
    73  					timer.Reset(debounce)
    74  				}
    75  			case err := <-watcher.Errors:
    76  				log.Println("error:", err)
    77  			}
    78  		}
    79  	}()
    80  	return nil
    81  }
