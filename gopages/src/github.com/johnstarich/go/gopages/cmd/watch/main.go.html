    36  func main() {
    37  	args, usageOutput, err := flags.Parse(os.Args[1:]...)
    38  	switch {
    39  	case err == nil:
    40  	case errors.Is(err, flag.ErrHelp):
    41  		fmt.Print(usageOutput)
    42  		return
    43  	default:
    44  		fmt.Print(usageOutput)
    45  		cmd.Exit(cmd.ExitCodeInvalidUsage)
    46  	}
    47  	args.Watch = true
    48  	var updatedTime string
    49  
    50  	ctx, cancel := context.WithCancel(context.Background())
    51  	defer cancel()
    52  
    53  	wd, err := os.Getwd()
    54  	if err != nil {
    55  		panic(err)
    56  	}
    57  	modulePackage, err := module.Package(wd)
    58  	if err != nil {
    59  		panic(err)
    60  	}
    61  	linker, err := args.Linker(modulePackage)
    62  	if err != nil {
    63  		panic(err)
    64  	}
    65  	fs, err := pagesFileSystem(ctx, wd, &updatedTime, args, linker)
    66  	if err != nil {
    67  		panic(err)
    68  	}
    69  	fileServer := http.FileServer(fs)
    70  
    71  	mux := http.NewServeMux()
    72  	mux.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
    73  		w.Header().Set(lastUpdatedHeader, updatedTime)
    74  
    75  		_, err := fs.Open(r.URL.Path)
    76  		if err == nil {
    77  			fileServer.ServeHTTP(w, r)
    78  			return
    79  		}
    80  		if !strings.HasSuffix(r.URL.Path, "/") {
    81  			// GitHub Pages automatically looks up a corresponding .html file if it exists
    82  			_, err := fs.Open(r.URL.Path + ".html")
    83  			if err == nil {
    84  				r.URL.Path += ".html"
    85  				fileServer.ServeHTTP(w, r)
    86  				return
    87  			}
    88  		}
    89  		r.URL.Path = "/404.html"
    90  		r.URL.RawPath = "/404.html"
    91  		fileServer.ServeHTTP(w, r)
    92  	})
    93  
    94  	server := http.Server{
    95  		Addr:              ":8080",
    96  		Handler:           mux,
    97  		ReadHeaderTimeout: time.Minute,
    98  	}
    99  	fmt.Println("Starting demo server on :8080...")
   100  	_ = server.ListenAndServe()
   101  }

   134  func pagesFileSystem(ctx context.Context, modulePath string, updateTime *string, args flags.Args, linker source.Linker) (http.FileSystem, error) {
   135  	out, err := pagesFSPipe.Do(pagesFSArgs{
   136  		Args:       args,
   137  		Ctx:        ctx,
   138  		ModulePath: modulePath,
   139  		UpdateTime: updateTime,
   140  		Linker:     linker,
   141  	})
   142  	var fs http.FileSystem
   143  	if err == nil {
   144  		fs = out[0].(http.FileSystem)
   145  	}
   146  	return fs, err

   154  func (h *httpFSWrapper) Open(name string) (http.File, error) {
   155  	if !strings.HasPrefix(name, h.base) {
   156  		return nil, os.ErrNotExist
   157  	}
   158  	name = strings.TrimPrefix(name, h.base)
   159  	name = filepath.FromSlash(name)
   160  	info, err := h.Stat(name)
   161  	if err != nil {
   162  		return nil, err
   163  	}
   164  
   165  	var file billy.File
   166  	// memfs.Open doesn't work for directories, so create a false dir for those instead
   167  	if info.IsDir() {
   168  		file, err = memfs.New().Create(name)
   169  	} else {
   170  		file, err = h.Filesystem.Open(name)
   171  	}
   172  	return &httpFileWrapper{
   173  		File: file,
   174  		fs:   h.Filesystem,
   175  		name: name,
   176  	}, err
   177  }
