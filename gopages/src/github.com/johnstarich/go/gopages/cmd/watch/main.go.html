    32  func main() {
    33  	args, usageOutput, err := flags.Parse(os.Args[1:]...)
    34  	switch err {
    35  	case nil:
    36  	case flag.ErrHelp:
    37  		fmt.Print(usageOutput)
    38  		return
    39  	default:
    40  		fmt.Print(usageOutput)
    41  		cmd.Exit(2)
    42  	}
    43  	args.Watch = true
    44  	var updatedTime string
    45  
    46  	ctx, cancel := context.WithCancel(context.Background())
    47  	defer cancel()
    48  
    49  	wd, err := os.Getwd()
    50  	if err != nil {
    51  		panic(err)
    52  	}
    53  	modulePackage, err := module.Package(wd)
    54  	if err != nil {
    55  		panic(err)
    56  	}
    57  	linker, err := args.Linker(modulePackage)
    58  	if err != nil {
    59  		panic(err)
    60  	}
    61  	fs, err := pagesFileSystem(ctx, wd, &updatedTime, args, linker)
    62  	if err != nil {
    63  		panic(err)
    64  	}
    65  	fileServer := http.FileServer(fs)
    66  
    67  	mux := http.NewServeMux()
    68  	mux.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
    69  		w.Header().Set(lastUpdatedHeader, updatedTime)
    70  
    71  		_, err := fs.Open(r.URL.Path)
    72  		if err == nil {
    73  			fileServer.ServeHTTP(w, r)
    74  			return
    75  		}
    76  		if !strings.HasSuffix(r.URL.Path, "/") {
    77  			// GitHub Pages automatically looks up a corresponding .html file if it exists
    78  			_, err := fs.Open(r.URL.Path + ".html")
    79  			if err == nil {
    80  				r.URL.Path += ".html"
    81  				fileServer.ServeHTTP(w, r)
    82  				return
    83  			}
    84  		}
    85  		r.URL.Path = "/404.html"
    86  		r.URL.RawPath = "/404.html"
    87  		fileServer.ServeHTTP(w, r)
    88  	})
    89  
    90  	server := http.Server{
    91  		Addr:    ":8080",
    92  		Handler: mux,
    93  	}
    94  	fmt.Println("Starting demo server on :8080...")
    95  	_ = server.ListenAndServe()
    96  }

   129  func pagesFileSystem(ctx context.Context, modulePath string, updateTime *string, args flags.Args, linker source.Linker) (http.FileSystem, error) {
   130  	out, err := pagesFSPipe.Do(pagesFSArgs{

   136  	})
   137  	var fs http.FileSystem

   149  func (h *httpFSWrapper) Open(name string) (http.File, error) {
   150  	if !strings.HasPrefix(name, h.base) {
   151  		return nil, os.ErrNotExist
   152  	}
   153  	name = strings.TrimPrefix(name, h.base)
   154  	name = filepath.FromSlash(name)
   155  	info, err := h.Filesystem.Stat(name)
   156  	if err != nil {
   157  		return nil, err
   158  	}
   159  
   160  	var file billy.File
   161  	// memfs.Open doesn't work for directories, so create a false dir for those instead
   162  	if info.IsDir() {
   163  		file, err = memfs.New().Create(name)
   164  	} else {
   165  		file, err = h.Filesystem.Open(name)
   166  	}
   167  	return &httpFileWrapper{
   168  		File: file,
   169  		fs:   h.Filesystem,
   170  		name: name,
   171  	}, err
   172  }
