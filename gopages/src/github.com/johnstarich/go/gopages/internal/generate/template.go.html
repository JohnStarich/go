   127  func nodeHTML(original nodeHTMLFunc, baseURL string, modulePackage string) nodeHTMLFunc {
   128  	// /pkg/mypkg/foo -> /base/pkg/mypkg/foo

   148  func linkReplacer(srcRoot, baseURL, modulePackage string) *strings.Replacer {
   149  	var (
   150  		defaultPackagePrefix = `<a href="` + srcRoot
   151  		defaultModulePrefix  = defaultPackagePrefix + modulePackage
   152  		gopagesModulePrefix  = fmt.Sprintf(`<a href="%s`, path.Join(baseURL, "/pkg", modulePackage))
   153  		publicDocPrefix      = `<a href="https://pkg.go.dev/`
   154  	)
   155  	return strings.NewReplacer(
   156  		defaultModulePrefix, gopagesModulePrefix, // /mypkg/foo -> /base/pkg/mypkg/foo
   157  		defaultPackagePrefix, publicDocPrefix, // /bar -> https://pkg.go.dev/bar
   158  	)
   159  }
