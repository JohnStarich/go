    25  	matches := reGoVersion.FindStringSubmatch(runtime.Version())
    26  	if len(matches) > 2 {

    28  		minor, _ := strconv.Atoi(matches[2]) // skip err check, minor will be 0 on error

    30  			return
    31  		}
