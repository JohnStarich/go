   351  func walkFilesFn(fs billy.Filesystem, path string, visit func(path string, isDir bool) error) error {
   352  	info, err := fs.Lstat(path)
   353  	if err != nil {
   354  		return errors.Wrap(err, "Error looking up file")
   355  	}
   356  
   357  	if info.Mode()&os.ModeSymlink == os.ModeSymlink {
   358  		// ignore symlinks to avoid infinite recursion
   359  		return nil
   360  	}
   361  
   362  	isDir := info.IsDir()
   363  	if err := visit(path, isDir); err != nil {
   364  		return err
   365  	}
   366  
   367  	if !isDir {
   368  		return nil
   369  	}
   370  
   371  	dir, err := fs.ReadDir(path)
   372  	if err != nil {
   373  		return errors.Wrapf(err, "Error reading directory %q", path)
   374  	}
   375  	for _, info = range dir {
   376  		err := walkFilesFn(fs, filepath.Join(path, info.Name()), visit)
   377  		if err == filepath.SkipDir {
   378  			if !info.IsDir() {
   379  				break // for SkipDir on a file, skip remaining files in directory
   380  			}
   381  			// otherwise continue recursing other files in this dir
   382  		} else if err != nil {
   383  			return err
   384  		}
   385  	}
   386  	return nil
   387  }
