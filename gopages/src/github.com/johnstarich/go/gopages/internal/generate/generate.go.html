   293  			}
   294  			_, err := pipe.Map(genStaticPipe, multiArgs)
   295  			return args, err

   297  		Concat(crawlerStaticPipe).
   298  		Concat(crawlerPipe).
   299  		Append(func(args docsArgs, packagePaths []string) (docsArgs, error) {
   300  			multiArgs := make([][]interface{}, len(packagePaths))
   301  			for i, p := range packagePaths {
   302  				multiArgs[i] = []interface{}{args, p}
   303  			}
   304  			_, err := pipe.Map(packageCrawlerPipe, multiArgs)
   305  			return args, err
   306  		}).
   307  		Concat(crawlerWalkPipe)
   308  	_, err := docsPipe.Do(dArgs)

   480  func walkFilesFn(fs billy.Filesystem, path string, visit func(path string, isDir bool) error) error {
   481  	info, err := fs.Lstat(path)
   482  	if err != nil {
   483  		return errors.Wrap(err, "Error looking up file")
   484  	}
   485  
   486  	if info.Mode()&os.ModeSymlink == os.ModeSymlink {
   487  		// ignore symlinks to avoid infinite recursion
   488  		return nil
   489  	}
   490  
   491  	isDir := info.IsDir()
   492  	if err := visit(path, isDir); err != nil {
   493  		return err
   494  	}
   495  
   496  	if !isDir {
   497  		return nil
   498  	}
   499  
   500  	dir, err := fs.ReadDir(path)
   501  	if err != nil {
   502  		return errors.Wrapf(err, "Error reading directory %q", path)
   503  	}
   504  	for _, info = range dir {
   505  		err := walkFilesFn(fs, filepath.Join(path, info.Name()), visit)
   506  		if errors.Is(err, filepath.SkipDir) {
   507  			if !info.IsDir() {
   508  				break // for SkipDir on a file, skip remaining files in directory
   509  			}
   510  			// otherwise continue recursing other files in this dir
   511  		} else if err != nil {
   512  			return err
   513  		}
   514  	}
   515  	return nil
   516  }
