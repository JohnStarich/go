    56  func (d *decoder) decodeGo(depth int) (result interface{}, endedBeforeDecode bool, err error) {
    57  	elem, foundEnd, err := d.decodeStartElement()
    58  	if foundEnd || err != nil {
    59  		return nil, foundEnd, err
    60  	}
    61  	logPad := strings.Repeat("  ", depth)
    62  	d.logger.Print(logPad, "start ", elem.Name.Local, "\n")
    63  	defer func() {
    64  		d.logger.Print(logPad, "end ", elem.Name.Local, ": ", result, "\n")
    65  		if err != nil {
    66  			d.logger.Println(" err: ", err)
    67  		}
    68  	}()
    69  
    70  	switch elem.Name.Local {
    71  	case "plist":
    72  		return nil, false, errFoundPList
    73  	case "array":
    74  		var elems []interface{}
    75  		for {
    76  			elem, endedBeforeDecode, err := d.decodeGo(depth + 1)
    77  			if endedBeforeDecode || err != nil {
    78  				return elems, false, err
    79  			}
    80  			elems = append(elems, elem)
    81  		}
    82  	case "dict":
    83  		elems := make(map[string]interface{})
    84  		for {
    85  			key, value, foundEnd, err := d.decodeKeyValue(depth + 1)
    86  			if foundEnd || err != nil {
    87  				return elems, false, err
    88  			}
    89  			elems[key] = value
    90  		}
    91  	case "true":
    92  		return true, false, d.xmlDecoder.Skip()
    93  	case "false":
    94  		return false, false, d.xmlDecoder.Skip()
    95  	case "integer":
    96  		var value int64
    97  		err := d.xmlDecoder.DecodeElement(&value, &elem)
    98  		return value, false, err
    99  	case "real":
   100  		var value float64
   101  		err := d.xmlDecoder.DecodeElement(&value, &elem)
   102  		return value, false, err
   103  	case "string":
   104  		var value string
   105  		err := d.xmlDecoder.DecodeElement(&value, &elem)
   106  		return value, false, err
   107  	case "data":
   108  		var value string
   109  		err := d.xmlDecoder.DecodeElement(&value, &elem)
   110  		if err != nil {
   111  			return nil, false, err
   112  		}
   113  		decoded, err := base64.StdEncoding.DecodeString(value)
   114  		return decoded, false, err
   115  	case "date":
   116  		var value string
   117  		err := d.xmlDecoder.DecodeElement(&value, &elem)
   118  		if err != nil {
   119  			return nil, false, err
   120  		}
   121  		parsed, err := time.Parse(time.RFC3339, value)

   123  	default:

   144  func (d *decoder) decodeKeyValue(depth int) (key string, value interface{}, foundEnd bool, err error) {
   145  	keyElem, foundEnd, err := d.decodeStartElement()
   146  	if foundEnd || err != nil {
   147  		return "", nil, foundEnd, err
   148  	}
   149  	err = d.xmlDecoder.DecodeElement(&key, &keyElem)
   150  	if err != nil {
   151  		return "", nil, false, err
   152  	}
   153  	value, endedBeforeDecode, err := d.decodeGo(depth)
   154  	if endedBeforeDecode || err != nil {
   155  		return "", nil, endedBeforeDecode, err
   156  	}
   157  	return key, value, false, nil
