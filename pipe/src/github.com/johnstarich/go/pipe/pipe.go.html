    38  func (p Pipe) Do(args ...interface{}) ([]interface{}, error) {
    39  	argVals := []reflect.Value{reflect.ValueOf(args)}
    40  	var errs []error
    41  	for _, op := range p.ops {

    43  		argVals, err = splitErrValue(op.Call(argVals))
    44  		if err != nil {
    45  			errs = append(errs, err)
    46  			if !p.options.KeepGoing {
    47  				break
    48  			}
    49  		}
    50  	}
    51  	if len(errs) > 0 {
    52  		return nil, Error{errs: errs}
    53  	}
    54

    56  	results := make([]interface{}, len(resultVals))
    57  	for i := range resultVals {
    58  		results[i] = resultVals[i].Interface()
    59  	}

    76  func (p Pipe) Concat(other Pipe) Pipe {
    77  	if len(p.ops) == 0 {
    78  		return other
    79  	}
    80  
    81  	lastOp := p.ops[len(p.ops)-1].Type()
    82  	out := make([]reflect.Type, lastOp.NumOut())
    83  	for i := range out {
    84  		out[i] = lastOp.Out(i)
    85  	}
    86  	if len(out) > 0 && isErr(out[len(out)-1]) {
    87  		// trim off last error return type, if present
    88  		out = out[:len(out)-1]
    89  	}
    90  	bridgeType := reflect.FuncOf(out, []reflect.Type{interfaceSliceType}, false)
    91  	bridgeFn := func(argVals []reflect.Value) (results []reflect.Value) {
    92  		args := make([]interface{}, len(argVals))
    93  		for i := range args {
    94  			args[i] = argVals[i].Interface()
    95  		}
    96  		return []reflect.Value{reflect.ValueOf(args)}
    97  	}
    98  
    99  	p = p.Append(reflect.MakeFunc(bridgeType, bridgeFn).Interface())
   100  	p.ops = append(p.ops, other.ops...)
   101  	return p

   104  func (p Pipe) appendFunc(fn interface{}) (Pipe, error) {
   105  	op := reflect.ValueOf(fn)
   106  	if op.Kind() != reflect.Func {
   107  		return p, fmt.Errorf("pipe value must a function, got: %T", fn)
   108  	}
   109  	if len(p.ops) == 0 {
   110  		opType := op.Type()
   111  		if opType.NumIn() != 1 || opType.In(0) != interfaceSliceType {
   112  			return p, fmt.Errorf("first pipe must accept 1 parameter of type []interface{}")
   113  		}
   114  	} else {
   115  		lastOp := p.ops[len(p.ops)-1]
   116  		if err := outMatchesIn(lastOp.Type(), op.Type()); err != nil {
   117  			return p, err
   118  		}
   119  	}
   120  	p.ops = append(p.ops, op)
   121  	return p, nil
   122  }

   124  func splitErrValue(args []reflect.Value) ([]reflect.Value, error) {
   125  	if len(args) == 0 {
   126  		return args, nil
   127  	}
   128  	lastVal := args[len(args)-1]
   129  	if !isErr(lastVal.Type()) {
   130  		return args, nil
   131  	}
   132  
   133  	errInt := lastVal.Interface()
   134  	var err error
   135  	if errInt != nil {
   136  		err = errInt.(error)
   137  	}
   138  	return args[:len(args)-1], err
   139  }

   145  func outMatchesIn(outFn, inFn reflect.Type) error {
   146  	var outTypes []reflect.Type
   147  	for i := 0; i < outFn.NumOut(); i++ {
   148  		outTypes = append(outTypes, outFn.Out(i))
   149  	}
   150  	var inTypes []reflect.Type
   151  	for i := 0; i < inFn.NumIn(); i++ {
   152  		inTypes = append(inTypes, inFn.In(i))
   153  	}
   154  
   155  	if len(outTypes) == len(inTypes)+1 && isErr(outTypes[len(outTypes)-1]) {
   156  		outTypes = outTypes[:len(outTypes)-1]
   157  	}
   158  
   159  	if len(outTypes) != len(inTypes) {
   160  		return fmt.Errorf("new function's parameter types do not match output function's return types: %v != %v", outTypes, inTypes)
   161  	}
   162  
   163  	for i := range outTypes {
   164  		out := outTypes[i]
   165  		in := inTypes[i]
   166  		if out != in {
   167  			return fmt.Errorf("new function's parameter type %v does not match the expected return type %v", in, out)
   168  		}
   169  	}
   170  	return nil
   171  }
