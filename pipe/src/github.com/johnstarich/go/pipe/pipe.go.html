    50  			}

    52  	}

    58  	results := make([]interface{}, len(resultVals))

    60  		results[i] = resultVals[i].Interface()

    78  func (p Pipe) Concat(other Pipe) Pipe {
    79  	if len(p.ops) == 0 {
    80  		return other
    81  	}
    82  
    83  	lastOp := p.ops[len(p.ops)-1].Type()
    84  	out := make([]reflect.Type, lastOp.NumOut())
    85  	for i := range out {
    86  		out[i] = lastOp.Out(i)
    87  	}
    88  	if len(out) > 0 && isErr(out[len(out)-1]) {
    89  		// trim off last error return type, if present
    90  		out = out[:len(out)-1]
    91  	}
    92  	bridgeType := reflect.FuncOf(out, []reflect.Type{interfaceSliceType}, false)
    93  	bridgeFn := func(argVals []reflect.Value) (results []reflect.Value) {
    94  		args := make([]interface{}, len(argVals))
    95  		for i := range args {
    96  			args[i] = argVals[i].Interface()
    97  		}
    98  		return []reflect.Value{reflect.ValueOf(args)}
    99  	}
   100  
   101  	p = p.Append(reflect.MakeFunc(bridgeType, bridgeFn).Interface())
   102  	p.ops = append(p.ops, other.ops...)
   103  	return p
   104  }

   106  func (p Pipe) appendFunc(fn interface{}) (Pipe, error) {
   107  	op := reflect.ValueOf(fn)
   108  	if op.Kind() != reflect.Func {
   109  		return p, fmt.Errorf("pipe value must a function, got: %T", fn)
   110  	}
   111  	if len(p.ops) == 0 {
   112  		opType := op.Type()
   113  		if opType.NumIn() != 1 || opType.In(0) != interfaceSliceType {
   114  			return p, fmt.Errorf("first pipe must accept 1 parameter of type []interface{}")
   115  		}
   116  	} else {
   117  		lastOp := p.ops[len(p.ops)-1]
   118  		if err := outMatchesIn(lastOp.Type(), op.Type()); err != nil {
   119  			return p, err
   120  		}
   121  	}
   122  	p.ops = append(p.ops, op)
   123  	return p, nil
   124  }

   126  func splitErrValue(args []reflect.Value) ([]reflect.Value, error) {
   127  	if len(args) == 0 {
   128  		return args, nil
   129  	}
   130  	lastVal := args[len(args)-1]
   131  	if !isErr(lastVal.Type()) {
   132  		return args, nil
   133  	}
   134  
   135  	errInt := lastVal.Interface()
   136  	var err error
   137  	if errInt != nil {
   138  		err = errInt.(error)
   139  	}
   140  	return args[:len(args)-1], err
   141  }

   147  func outMatchesIn(outFn, inFn reflect.Type) error {
   148  	var outTypes []reflect.Type
   149  	for i := 0; i < outFn.NumOut(); i++ {
   150  		outTypes = append(outTypes, outFn.Out(i))
   151  	}
   152  	var inTypes []reflect.Type
   153  	for i := 0; i < inFn.NumIn(); i++ {
   154  		inTypes = append(inTypes, inFn.In(i))
   155  	}
   156  
   157  	if len(outTypes) == len(inTypes)+1 && isErr(outTypes[len(outTypes)-1]) {
   158  		outTypes = outTypes[:len(outTypes)-1]
   159  	}
   160  
   161  	if len(outTypes) != len(inTypes) {
   162  		return fmt.Errorf("new function's parameter types do not match output function's return types: %v != %v", outTypes, inTypes)
   163  	}
   164  
   165  	for i := range outTypes {
   166  		out := outTypes[i]
   167  		in := inTypes[i]
   168  		if out != in {
   169  			return fmt.Errorf("new function's parameter type %v does not match the expected return type %v", in, out)
   170  		}
   171  	}
   172  	return nil
   173  }
