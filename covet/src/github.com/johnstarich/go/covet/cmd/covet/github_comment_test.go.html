    16  func TestEnsureAppGitHubComment(t *testing.T) {
    17  	t.Parallel()
    18  	someCommentID := int64(1)
    19  	created := false
    20  	updated := false
    21  	server := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
    22  		switch fmt.Sprintf("%s:%s", r.Method, r.URL.Path) {
    23  		// create new comment
    24  		case "GET:/api/v3/repos/my-bot/org/myrepo/issues/456/comments":
    25  			require.NoError(t, json.NewEncoder(w).Encode([]*github.IssueComment{}))
    26  		case "POST:/api/v3/repos/my-bot/org/myrepo/issues/456/comments":
    27  			created = true
    28  			var comment github.IssueComment
    29  			require.NoError(t, json.NewDecoder(r.Body).Decode(&comment))
    30  			assert.Equal(t, github.IssueComment{
    31  				Body: stringPtr(`<!-- covet -->
    32  
    33  some body`),
    34  			}, comment)
    35  			w.WriteHeader(http.StatusNoContent)
    36  
    37  		// update existing comment
    38  		case "GET:/api/v3/repos/my-bot/org/myrepo/issues/123/comments":
    39  			require.NoError(t, json.NewEncoder(w).Encode([]*github.IssueComment{
    40  				{
    41  					ID: &someCommentID,
    42  					Body: stringPtr(`<!-- covet -->
    43  
    44  some body`),
    45  				},
    46  			}))
    47  		case "PATCH:/api/v3/repos/my-bot/org/myrepo/issues/comments/1":
    48  			updated = true
    49  			var comment github.IssueComment
    50  			require.NoError(t, json.NewDecoder(r.Body).Decode(&comment))
    51  			assert.Equal(t, github.IssueComment{
    52  				Body: stringPtr(`<!-- covet -->
    53  
    54  some other body`),
    55  			}, comment)
    56  			w.WriteHeader(http.StatusNoContent)
    57  		default:
    58  			t.Fatal("Unknown request method and path:", r.Method, r.URL.Path)
    59  		}
    60  	}))
    61  	t.Cleanup(server.Close)
    62  
    63  	ctx := context.Background()
    64  	// create new comment
    65  	err := ensureAppGitHubComment(ctx, gitHubCommentOptions{
    66  		GitHubEndpoint: server.URL,
    67  		GitHubToken:    "some-token",
    68  		RepoOwner:      "my-bot",
    69  		Repo:           "org/myrepo",
    70  		IssueNumber:    456,
    71  		Body:           "some body",
    72  	})
    73  	assert.NoError(t, err)
    74  	assert.True(t, created)
    75  	// update existing comment
    76  	err = ensureAppGitHubComment(ctx, gitHubCommentOptions{
    77  		GitHubEndpoint: server.URL,
    78  		GitHubToken:    "some-token",
    79  		RepoOwner:      "my-bot",
