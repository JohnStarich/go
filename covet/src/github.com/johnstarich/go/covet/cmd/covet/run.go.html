   129  func runArgs(args Args, deps Deps) (err error) {
   130  	defer func() { err = errors.WithStack(err) }()
   131  
   132  	var diffFile io.Reader
   133  	if args.DiffFile == "-" {
   134  		diffFile = deps.Stdin
   135  	} else {
   136  		f, err := deps.FS.Open(args.DiffFile)
   137  		if err != nil {
   138  			return err
   139  		}
   140  		defer f.Close()
   141  		diffFile = f
   142  	}
   143  
   144  	cov, err := covet.Parse(covet.Options{
   145  		FS:             deps.FS,
   146  		Diff:           diffFile,
   147  		DiffBaseDir:    args.DiffBaseDir,
   148  		GoCoveragePath: args.GoCoverageFile,
   149  	})
   150  	if err != nil {
   151  		return err
   152  	}
   153  	if len(cov.DiffCoverageFiles()) == 0 {
   154  		fmt.Fprintln(deps.Stdout, "No coverage information intersects with diff.")
   155  		return nil
   156  	}
   157  
   158  	uncoveredFiles := cov.PriorityUncoveredFiles(args.TargetDiffCoverage)
   159  	if args.ShowCoverage {
   160  		for _, f := range uncoveredFiles {
   161  			fmt.Fprintln(deps.Stdout, "Coverage diff:", f.Name)
   162  			err := cov.ReportFileCoverage(deps.Stdout, f, covet.ReportFileCoverageOptions{})
   163  			if err != nil {
   164  				return err
   165  			}
   166  		}
   167  	}
   168  
   169  	fmt.Fprintln(deps.Stdout)
   170  	totalCovered := cov.DiffCovered()
   171  	totalCoveredStatus := coverstatus.New(totalCovered)
   172  	fmt.Fprintln(deps.Stdout, "Total diff coverage:", totalCoveredStatus.Colorize(summary.FormatPercent(totalCovered)))
   173  	fmt.Fprintln(deps.Stdout)
   174  	summaryOptions := covet.ReportSummaryOptions{Target: args.TargetDiffCoverage}
   175  	err = cov.ReportSummaryColorTerminal(deps.Stdout, summaryOptions)
   176  	if err != nil {
   177  		return err
   178  	}
   179  
   180  	runWorkflow(coverageCommand(totalCovered, "", nil))
   181  	for _, f := range uncoveredFiles {
   182  		runWorkflow(coverageCommand(summary.FileCoverage(f), f.Name, findUncoveredLines(f)))
   183  	}
   184  
   185  	if args.GitHubToken != "" {
   186  		org, repo, number, err := parseIssueURL(args.GitHubIssue)
   187  		if err != nil {
   188  			return err
   189  		}
   190  		var markdownReport bytes.Buffer
   191  		err = cov.ReportSummaryMarkdown(&markdownReport, summaryOptions)
   192  		if err != nil {
   193  			return err
   194  		}
   195  		err = ensureAppGitHubComment(context.Background(), gitHubCommentOptions{
   196  			GitHubEndpoint: args.GitHubEndpoint,
   197  			GitHubToken:    args.GitHubToken,
   198  			RepoOwner:      org,
   199  			Repo:           repo,
   200  			IssueNumber:    number,
   201  			Body:           markdownReport.String(),
   202  		})
   203  		if err != nil {
   204  			fmt.Fprintln(deps.Stdout, "\nFailed to update GitHub comment, skipping. Error:", err)
   205  		}
   206  	}
   207  	return nil
   208  }

   210  func findUncoveredLines(f covet.File) []span.Span {
   211  	var uncoveredLines []span.Span
   212  	ok := true
   213  	var nextLineIndex int
   214  	for ok {
   215  		var uncovered span.Span
   216  		uncovered, ok, nextLineIndex = findFirstUncoveredLines(f.Lines, nextLineIndex)
   217  		if ok {
   218  			uncoveredLines = append(uncoveredLines, uncovered)
   219  		}
   220  	}
   221  	sort.SliceStable(uncoveredLines, func(a, b int) bool {
   222  		return uncoveredLines[a].Len() > uncoveredLines[b].Len()
   223  	})
   224  	return uncoveredLines
   225  }

   227  func findFirstUncoveredLines(lines []covet.Line, startIndex int) (uncovered span.Span, ok bool, nextLineIndex int) {
   228  	// find start
   229  	nextLineIndex = startIndex
   230  	for _, l := range lines[nextLineIndex:] {
   231  		nextLineIndex++
   232  		if !l.Covered {
   233  			n := l.LineNumber
   234  			uncovered = span.Span{
   235  				Start: n,
   236  				End:   n + 1,
   237  			}
   238  			ok = true
   239  			break
   240  		}
   241  	}
   242  	// find next line number jump or covered line
   243  	for _, l := range lines[nextLineIndex:] {
   244  		if l.Covered || l.LineNumber != uncovered.End {
   245  			break
   246  		}
   247  		nextLineIndex++
   248  		uncovered.End++
   249  	}
   250  	return
   251  }

   258  func parseIssueURL(s string) (org, repo string, number int, err error) {
   259  	if s == "" {
   260  		err = fmt.Errorf("-gh-issue is required")
   261  		return
   262  	}
   263  	issueURL, err := url.Parse(s)
   264  	if err != nil {
   265  		return
   266  	}
   267  	if issueURL.Scheme == "" {
   268  		issueURL, err = url.Parse("https://" + s)
   269  		if err != nil {
   270  			return
   271  		}
   272  	}
   273  	const minIssueURLPathComponents = 4
   274  	tokens := strings.SplitN(strings.TrimPrefix(issueURL.Path, "/"), "/", minIssueURLPathComponents+1)
   275  	if len(tokens) < minIssueURLPathComponents {
   276  		err = fmt.Errorf("malformed issue URL: expected 4+ path components, e.g. github.com/org/repo/pull/123")
   277  		return
   278  	}
   279  	n, err := strconv.ParseInt(tokens[3], decimalBase, maxIntBits)
   280  	if err != nil {
   281  		return
   282  	}
   283  	org = tokens[0]
   284  	repo = tokens[1]
   285  	number = int(n)
   286  	return
   287  }
