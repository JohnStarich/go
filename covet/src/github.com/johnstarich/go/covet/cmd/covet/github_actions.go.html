    42  func githubActionsEncode(s string) string {
    43  	var sb strings.Builder
    44  	for _, r := range s {
    45  		switch r {
    46  		case '%':
    47  			sb.WriteString("%25")
    48  		case '\r':
    49  			sb.WriteString("%0D")
    50  		case '\n':
    51  			sb.WriteString("%0A")
    52  		case ':':
    53  			sb.WriteString("%3A")
    54  		case ',':
    55  			sb.WriteString("%2C")
    56  		default:
    57  			sb.WriteRune(r)
    58  		}
    59  	}
    60  	return sb.String()
