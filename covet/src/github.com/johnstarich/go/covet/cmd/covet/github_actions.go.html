    15  	var sb strings.Builder

    32  	}

    40  // Mapping guide found here: https://pakstech.com/blog/github-actions-workflow-commands/#what-are-workflow-commands
    41  func githubActionsEncode(s string) string {
    42  	var sb strings.Builder
    43  	for _, r := range s {
    44  		switch r {
    45  		case '%':
    46  			sb.WriteString("%25")
    47  		case '\r':
    48  			sb.WriteString("%0D")
    49  		case '\n':
    50  			sb.WriteString("%0A")
    51  		case ':':
    52  			sb.WriteString("%3A")
    53  		case ',':
    54  			sb.WriteString("%2C")
    55  		default:
    56  			sb.WriteRune(r)
    57  		}
    58  	}
    59  	return sb.String()
    60  }
