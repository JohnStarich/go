    47  func Parse(options Options) (covet *Covet, err error) {
    48  	defer func() { err = errors.Wrap(err, "covet") }()
    49  	if !hackpadfs.ValidPath(options.DiffBaseDir) {
    50  		return nil, errors.Errorf("invalid diff base directory FS path: %s", options.DiffBaseDir)
    51  	}
    52  	if !hackpadfs.ValidPath(options.GoCoveragePath) {
    53  		return nil, errors.Errorf("invalid coverage FS path: %s", options.GoCoveragePath)
    54  	}
    55  	if options.GoCoverageBaseDir == "" {
    56  		options.GoCoverageBaseDir = path.Dir(options.GoCoveragePath)
    57  	}
    58  	if !hackpadfs.ValidPath(options.GoCoverageBaseDir) {
    59  		return nil, errors.Errorf("invalid coverage base directory FS path: %s", options.GoCoverageBaseDir)
    60  	}
    61  	if options.FS == nil {
    62  		options.FS, err = fspath.WorkingDirectoryFS()
    63  		if err != nil {
    64  			return nil, err
    65  		}
    66  	}
    67  	if options.Diff == nil {
    68  		return nil, errors.New("diff reader must not be nil")
    69  	}
    70  
    71  	diffFiles, _, err := gitdiff.Parse(options.Diff)
    72  	if err != nil {
    73  		return nil, err
    74  	}
    75  
    76  	coverageFile, err := options.FS.Open(options.GoCoveragePath)
    77  	if err != nil {
    78  		return nil, err
    79  	}
    80  	defer coverageFile.Close()
    81  	coverageFiles, err := cover.ParseProfilesFromReader(coverageFile)
    82  	if err != nil {
    83  		return nil, err
    84  	}
    85  
    86  	covet = &Covet{
    87  		options:        options,
    88  		addedLines:     make(map[string][]span.Span),
    89  		coveredLines:   make(map[string][]span.Span),
    90  		uncoveredLines: make(map[string][]span.Span),
    91  	}
    92  	_, err = covet.coverageToDiffRel()
    93  	if err != nil {
    94  		return nil, err
    95  	}
    96  
    97  	covet.addDiff(diffFiles)
    98  	if err := covet.addCoverage(options.FS, options.GoCoverageBaseDir, coverageFiles); err != nil {
    99  		return nil, err
   100  	}
   101  	return covet, nil
   102  }

   116  // If i is outside 0 to [math.MaxUint32], then it is capped at those bounds.
   117  func uintFromBoundedSignedInt[integer signedInteger](i integer) uint {
   118  	if i < 0 {
   119  		return 0
   120  	}
   121  	if i > math.MaxUint32 {
   122  		return math.MaxUint32
   123  	}
   124  	return uint(i)

   127  func findDiffAddSpans(fragments []*gitdiff.TextFragment) []span.Span {
   128  	var spans []span.Span
   129  	for _, fragment := range fragments {
   130  		lineNumber := uintFromBoundedSignedInt(fragment.NewPosition)
   131  		for _, line := range fragment.Lines {
   132  			if line.Op == gitdiff.OpAdd {
   133  				if len(spans) == 0 || spans[len(spans)-1].End < lineNumber {
   134  					spans = append(spans, span.Span{Start: lineNumber, End: lineNumber + 1})
   135  				} else {
   136  					spans[len(spans)-1].End++
   137  				}
   138  			}
   139  			if line.New() {
   140  				lineNumber++
   141  			}
   142  		}
   143  	}
   144  	return spans
   145  }

   147  func (c *Covet) addCoverage(fs hackpadfs.FS, baseDir string, coverageFiles []*cover.Profile) error {
   148  	for _, file := range coverageFiles {
   149  		for _, block := range file.Blocks {
   150  			coverageFile, err := packages.FilePath(fs, baseDir, file.FileName, packages.Options{})
   151  			if err != nil {
   152  				return err
   153  			}
   154  			if block.Count > 0 {
   155  				c.coveredLines[coverageFile] = append(c.coveredLines[coverageFile], span.Span{
   156  					Start: uintFromBoundedSignedInt(block.StartLine),
   157  					End:   uintFromBoundedSignedInt(block.EndLine + 1),
   158  				})
   159  			} else {
   160  				c.uncoveredLines[coverageFile] = append(c.uncoveredLines[coverageFile], span.Span{
   161  					Start: uintFromBoundedSignedInt(block.StartLine),
   162  					End:   uintFromBoundedSignedInt(block.EndLine + 1),
   163  				})
   164  			}
   165  		}
   166  	}
   167  	return nil
   168  }

   174  func (c *Covet) coveredAndUncovered() (fileNames map[string]bool, coveredDiff, uncoveredDiff map[string][]span.Span) {
   175  	covToDiffRel, _ := c.coverageToDiffRel() // ignore error since it's checked during setup
   176  	fileNames = make(map[string]bool)
   177  	coveredDiff = make(map[string][]span.Span)
   178  	uncoveredDiff = make(map[string][]span.Span)
   179  	for file := range c.addedLines {
   180  		for _, added := range c.addedLines[file] {
   181  			file, err := fspath.Rel(covToDiffRel, file)
   182  			if err != nil {
   183  				panic(err)
   184  			}
   185  			for _, covered := range c.coveredLines[file] {
   186  				if intersection, ok := added.Intersection(covered); ok {
   187  					fileNames[file] = true
   188  					coveredDiff[file] = append(coveredDiff[file], intersection)
   189  				}
   190  			}
   191  			for _, uncovered := range c.uncoveredLines[file] {
   192  				if intersection, ok := added.Intersection(uncovered); ok {
   193  					fileNames[file] = true
   194  					uncoveredDiff[file] = append(uncoveredDiff[file], intersection)
   195  				}
   196  			}
   197  		}
   198  	}
   199  	return
   200  }

   202  // DiffCovered returns the percentage of covered lines in the diff.
   203  func (c *Covet) DiffCovered() float64 {
   204  	_, coveredSpans, uncoveredSpans := c.coveredAndUncovered()
   205  	var coveredTotal, uncoveredTotal float64
   206  	for _, spans := range coveredSpans {
   207  		for _, s := range spans {
   208  			coveredTotal += float64(s.Len())
   209  		}
   210  	}
   211  	for _, spans := range uncoveredSpans {
   212  		for _, s := range spans {
   213  			uncoveredTotal += float64(s.Len())
   214  		}
   215  	}
   216  	return coveredTotal / (coveredTotal + uncoveredTotal)
   217  }

   219  // DiffCoverageFiles generates a report of all files that are in the diff and have coverage information.
   220  // These files can be used to display diff coverage information.
   221  func (c *Covet) DiffCoverageFiles() []File {
   222  	var coveredFiles []File
   223  	fileNames, coveredSpans, uncoveredSpans := c.coveredAndUncovered()
   224  	for file := range fileNames {
   225  		covered := coveredSpans[file]
   226  		uncovered := uncoveredSpans[file]
   227  
   228  		coveredFile := File{Name: file}
   229  		for _, s := range covered {
   230  			for i := s.Start; i < s.End; i++ {
   231  				coveredFile.Lines = append(coveredFile.Lines, Line{
   232  					Covered:    true,
   233  					LineNumber: i,
   234  				})
   235  			}
   236  			coveredFile.Covered += s.Len()
   237  		}
   238  		for _, s := range uncovered {
   239  			for i := s.Start; i < s.End; i++ {
   240  				coveredFile.Lines = append(coveredFile.Lines, Line{
   241  					Covered:    false,
   242  					LineNumber: i,
   243  				})
   244  			}
   245  			coveredFile.Uncovered += s.Len()
   246  		}
   247  		sort.Slice(coveredFile.Lines, func(a, b int) bool {
   248  			return coveredFile.Lines[a].LineNumber < coveredFile.Lines[b].LineNumber
   249  		})
   250  		coveredFiles = append(coveredFiles, coveredFile)
   251  	}
   252  	return coveredFiles
   253  }
