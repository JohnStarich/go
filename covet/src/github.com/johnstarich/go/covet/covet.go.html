    45  // Parse reads and parses both a diff file and Go coverage file, then returns a Covet instance to render reports
    46  func Parse(options Options) (covet *Covet, err error) {
    47  	defer func() { err = errors.Wrap(err, "covet") }()
    48  	if !hackpadfs.ValidPath(options.DiffBaseDir) {
    49  		return nil, errors.Errorf("invalid diff base directory FS path: %s", options.DiffBaseDir)
    50  	}
    51  	if !hackpadfs.ValidPath(options.GoCoveragePath) {
    52  		return nil, errors.Errorf("invalid coverage FS path: %s", options.GoCoveragePath)
    53  	}
    54  	if options.GoCoverageBaseDir == "" {
    55  		options.GoCoverageBaseDir = path.Dir(options.GoCoveragePath)
    56  	}
    57  	if !hackpadfs.ValidPath(options.GoCoverageBaseDir) {
    58  		return nil, errors.Errorf("invalid coverage base directory FS path: %s", options.GoCoverageBaseDir)
    59  	}
    60  	if options.FS == nil {
    61  		options.FS, err = fspath.WorkingDirectoryFS()
    62  		if err != nil {
    63  			return nil, err
    64  		}
    65  	}
    66  	if options.Diff == nil {
    67  		return nil, errors.New("diff reader must not be nil")
    68  	}
    69  
    70  	diffFiles, _, err := gitdiff.Parse(options.Diff)
    71  	if err != nil {
    72  		return nil, err
    73  	}
    74  
    75  	coverageFile, err := options.FS.Open(options.GoCoveragePath)
    76  	if err != nil {
    77  		return nil, err
    78  	}
    79  	defer coverageFile.Close()
    80  	coverageFiles, err := cover.ParseProfilesFromReader(coverageFile)
    81  	if err != nil {
    82  		return nil, err
    83  	}
    84  
    85  	covet = &Covet{
    86  		options:        options,
    87  		addedLines:     make(map[string][]span.Span),
    88  		coveredLines:   make(map[string][]span.Span),
    89  		uncoveredLines: make(map[string][]span.Span),
    90  	}
    91  	_, err = covet.coverageToDiffRel()
    92  	if err != nil {
    93  		return nil, err
    94  	}
    95  
    96  	covet.addDiff(diffFiles)
    97  	if err := covet.addCoverage(options.FS, options.GoCoverageBaseDir, coverageFiles); err != nil {
    98  		return nil, err
    99  	}
   100  	return covet, nil
   101  }

   103  func (c *Covet) addDiff(diffFiles []*gitdiff.File) {
   104  	for _, file := range diffFiles {
   105  		spans := findDiffAddSpans(file.TextFragments)
   106  		c.addedLines[file.NewName] = append(c.addedLines[file.NewName], spans...)

   110  func findDiffAddSpans(fragments []*gitdiff.TextFragment) []span.Span {
   111  	var spans []span.Span
   112  	for _, fragment := range fragments {
   113  		lineNumber := fragment.NewPosition
   114  		for _, line := range fragment.Lines {
   115  			if line.Op == gitdiff.OpAdd {
   116  				if len(spans) == 0 || spans[len(spans)-1].End < lineNumber {
   117  					spans = append(spans, span.Span{Start: lineNumber, End: lineNumber + 1})
   118  				} else {
   119  					spans[len(spans)-1].End++
   120  				}
   121  			}
   122  			if line.New() {
   123  				lineNumber++
   124  			}
   125  		}
   126  	}
   127  	return spans
   128  }

   130  func (c *Covet) addCoverage(fs hackpadfs.FS, baseDir string, coverageFiles []*cover.Profile) error {
   131  	for _, file := range coverageFiles {
   132  		for _, block := range file.Blocks {
   133  			coverageFile, err := packages.FilePath(fs, baseDir, file.FileName, packages.Options{})
   134  			if err != nil {
   135  				return err
   136  			}
   137  			if block.Count > 0 {
   138  				c.coveredLines[coverageFile] = append(c.coveredLines[coverageFile], span.Span{
   139  					Start: int64(block.StartLine),
   140  					End:   int64(block.EndLine + 1),
   141  				})
   142  			} else {
   143  				c.uncoveredLines[coverageFile] = append(c.uncoveredLines[coverageFile], span.Span{
   144  					Start: int64(block.StartLine),
   145  					End:   int64(block.EndLine + 1),
   146  				})
   147  			}
   148  		}
   149  	}
   150  	return nil
   151  }

   157  func (c *Covet) coveredAndUncovered() (fileNames map[string]bool, coveredDiff, uncoveredDiff map[string][]span.Span) {
   158  	covToDiffRel, _ := c.coverageToDiffRel() // ignore error since it's checked during setup
   159  	fileNames = make(map[string]bool)
   160  	coveredDiff = make(map[string][]span.Span)
   161  	uncoveredDiff = make(map[string][]span.Span)
   162  	for file := range c.addedLines {
   163  		for _, added := range c.addedLines[file] {
   164  			file, err := fspath.Rel(covToDiffRel, file)
   165  			if err != nil {
   166  				panic(err)
   167  			}
   168  			for _, covered := range c.coveredLines[file] {
   169  				if intersection, ok := added.Intersection(covered); ok {
   170  					fileNames[file] = true
   171  					coveredDiff[file] = append(coveredDiff[file], intersection)
   172  				}
   173  			}
   174  			for _, uncovered := range c.uncoveredLines[file] {
   175  				if intersection, ok := added.Intersection(uncovered); ok {
   176  					fileNames[file] = true
   177  					uncoveredDiff[file] = append(uncoveredDiff[file], intersection)
   178  				}
   179  			}
   180  		}
   181  	}
   182  	return
   183  }

   185  // DiffCovered returns the percentage of covered lines in the diff.
   186  func (c *Covet) DiffCovered() float64 {
   187  	_, coveredSpans, uncoveredSpans := c.coveredAndUncovered()
   188  	var coveredTotal, uncoveredTotal float64
   189  	for _, spans := range coveredSpans {
   190  		for _, s := range spans {
   191  			coveredTotal += float64(s.Len())
   192  		}
   193  	}
   194  	for _, spans := range uncoveredSpans {
   195  		for _, s := range spans {
   196  			uncoveredTotal += float64(s.Len())
   197  		}
   198  	}
   199  	return coveredTotal / (coveredTotal + uncoveredTotal)
   200  }

   202  // DiffCoverageFiles generates a report of all files that are in the diff and have coverage information.
   203  // These files can be used to display diff coverage information.
   204  func (c *Covet) DiffCoverageFiles() []File {
   205  	var coveredFiles []File
   206  	fileNames, coveredSpans, uncoveredSpans := c.coveredAndUncovered()
   207  	for file := range fileNames {
   208  		covered := coveredSpans[file]
   209  		uncovered := uncoveredSpans[file]
   210  
   211  		coveredFile := File{Name: file}
   212  		for _, s := range covered {
   213  			for i := s.Start; i < s.End; i++ {
   214  				coveredFile.Lines = append(coveredFile.Lines, Line{
   215  					Covered:    true,
   216  					LineNumber: uint(i),
   217  				})
   218  			}
   219  			coveredFile.Covered += uint(s.Len())
   220  		}
   221  		for _, s := range uncovered {
   222  			for i := s.Start; i < s.End; i++ {
   223  				coveredFile.Lines = append(coveredFile.Lines, Line{
   224  					Covered:    false,
   225  					LineNumber: uint(i),
   226  				})
   227  			}
   228  			coveredFile.Uncovered += uint(s.Len())
   229  		}
   230  		sort.Slice(coveredFile.Lines, func(a, b int) bool {
   231  			return coveredFile.Lines[a].LineNumber < coveredFile.Lines[b].LineNumber
   232  		})
   233  		coveredFiles = append(coveredFiles, coveredFile)
   234  	}
   235  	return coveredFiles
   236  }
