    43  func findReportableUncoveredFiles(coveredFiles []File, target, current float64) []File {
    44  	// sort by highest uncovered line count
    45  	sort.Slice(coveredFiles, func(aIndex, bIndex int) bool {
    46  		a, b := coveredFiles[aIndex], coveredFiles[bIndex]
    47  		switch {
    48  		case a.Uncovered != b.Uncovered:
    49  			return a.Uncovered > b.Uncovered
    50  		default:
    51  			return a.Name < b.Name
    52  		}
    53  	})
    54  
    55  	var uncoveredFiles []File
    56  	// find minimum number of covered lines required to hit target
    57  	targetMissingLines := uint(0)
    58  	totalLines := uint(0)
    59  	for _, f := range coveredFiles {
    60  		totalLines += f.Covered + f.Uncovered
    61  	}
    62  	if percentDiff := target - current; percentDiff > 0 {
    63  		targetMissingLines = uint(percentDiff * float64(totalLines))
    64  	} else {
    65  		return nil // target is met
    66  	}
    67  	// next, collect the biggest uncovered files until we'd hit the target
    68  	for _, f := range coveredFiles {
    69  		const minUncoveredThreshold = 2 // include more files if it is slim pickings
    70  		if f.Uncovered > 0 {
    71  			uncoveredFiles = append(uncoveredFiles, f)
    72  		}
    73  		if f.Uncovered > minUncoveredThreshold {
    74  			targetMissingLines -= f.Uncovered
    75  		}
    76  		if targetMissingLines <= 0 {
    77  			break
    78  		}
    79  	}
    80  	return uncoveredFiles
