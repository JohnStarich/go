    22  func (f File) findContextSpans(contextLines uint) []span.Span {
    23  	if len(f.Lines) == 0 {
    24  		return nil
    25  	}
    26  
    27  	var overlappingSpans []span.Span
    28  	for _, line := range f.Lines {
    29  		overlappingSpans = append(overlappingSpans, span.Span{
    30  			Start: max(1, int64(line.LineNumber)-int64(contextLines)),
    31  			End:   int64(line.LineNumber + contextLines + 1), // may be beyond end of file, but can stop line iteration at EOF
    32  		})
    33  	}
    34  	spans := []span.Span{overlappingSpans[0]}
    35  	for _, s := range overlappingSpans {
    36  		lastSpan := &spans[len(spans)-1]
    37  		newSpan, merged := lastSpan.Merge(s)
    38  		if merged {
    39  			*lastSpan = newSpan
    40  		} else {
    41  			spans = append(spans, s)
    42  		}
    43  	}
    44  	return spans
    45  }

    76  func DiffChunks(file File, fileReader io.Reader) ([]DiffChunk, error) {
    77  	var chunks []DiffChunk
    78  	iter := newLineIterator(fileReader)
    79  	const contextLines = 2
    80  	var lineNumber int64 = 1
    81  	diffLineIndex := 0
    82  	for _, s := range file.findContextSpans(contextLines) {
    83  		if lineNumber < s.Start {
    84  			err := iter.SkipLines(s.Start - lineNumber)
    85  			if err != nil {
    86  				return nil, err
    87  			}
    88  			lineNumber = s.Start
    89  		}
    90  		lines, err := iter.NextLines(s.Len())
    91  		if err != nil {
    92  			return nil, err
    93  		}
    94  		for i := range lines {
    95  			op := noOpPrefix
    96  			if diffLineIndex < len(file.Lines) {
    97  				diffLine := file.Lines[diffLineIndex]
    98  				if lineNumber == int64(diffLine.LineNumber) {
    99  					op = diffLine.diffOpPrefix()
   100  					diffLineIndex++
   101  				}
   102  			}
   103  			lines[i] = op + lines[i]
   104  			lineNumber++
   105  		}
   106  		chunks = append(chunks, DiffChunk{
   107  			FirstLine: uint(s.Start),
   108  			LastLine:  uint(lineNumber - 1),
   109  			Lines:     lines,
   110  		})
   111  	}
   112  	return chunks, nil
   113  }
