    29  //
    30  // Pass a file package pattern like "github.com/org/mymodule/myfile.go" or "./mymodule/myfile.go".

    32  func FilePath(fs hackpadfs.FS, workingDirectory, filePattern string, options Options) (pkgFile string, err error) {
    33  	memFS, err := mem.NewFS()
    34  	panicIfErr(err)
    35  	mountFS, err := mount.NewFS(memFS)
    36  	panicIfErr(err)
    37  
    38  	const (
    39  		dirPerm = 0700
    40  		workDir = "work"
    41  		tempDir = "tmp"
    42  	)

    44  	setErr(memFS.Mkdir(workDir, dirPerm), &sErr)
    45  	setErr(mountFS.AddMount(workDir, fs), &sErr)

    49  
    50  	moduleName, moduleDir, err := findModule(mountFS, workingDirectory)
    51  	setErr(err, &sErr)
    52  	trimDir := workingDirectory
    53  	if moduleName != "" {
    54  		memFS, err := mem.NewFS()
    55  		panicIfErr(err)
    56  		setErr(mountFS.AddMount(tempDir, memFS), &sErr)
    57  		moduleGoPath := path.Join("src", moduleName)
    58  		const dirPermission = 0700
    59  		setErr(memFS.MkdirAll(moduleGoPath, dirPermission), &sErr)
    60  		moduleFS, err := hackpadfs.Sub(mountFS, moduleDir)

    63  		}
    64  		setErr(mountFS.AddMount(path.Join(tempDir, moduleGoPath), moduleFS), &sErr) // "symlink" original fs inside the new GOPATH-like directory

    66  		workDirSubPath = strings.TrimPrefix(workDirSubPath, "/")

    69

    71  	ctx := newFSBuildContext(mountFS, options)
    72  	pkg, err := ctx.Import(packageName, workingDirectory, build.FindOnly)
    73  	setErr(err, &sErr)
    74  	coverageFile = path.Join(pkg.Dir, coverageFile)
    75  	coverageFile = strings.TrimPrefix(coverageFile, trimDir+"/")

    99  func newFSBuildContext(fs hackpadfs.FS, options Options) build.Context {
   100  	ctx := build.Default
   101  	ctx.GOROOT = options.GoRoot
   102  	ctx.GOPATH = options.GoPath
   103  	ctx.JoinPath = path.Join
   104  	ctx.SplitPathList = filepath.SplitList
   105  	ctx.IsAbsPath = func(path string) bool {
   106  		return !build.IsLocalImport(path)
   107  	}
   108  	ctx.IsDir = func(path string) bool {
   109  		info, err := hackpadfs.Stat(fs, path)
   110  		return err == nil && info.IsDir()
   111  	}
   112  	ctx.HasSubdir = func(root, dir string) (rel string, ok bool) {
   113  		// TODO add EvalSymlinks support to hackpadfs
   114  		const sep = "/"
   115  		root = path.Clean(root)
   116  		if !strings.HasSuffix(root, sep) {
   117  			root += sep
   118  		}
   119  		dir = path.Clean(dir)
   120  		if !strings.HasPrefix(dir, root) {
   121  			return "", false
   122  		}
   123  		return dir[len(root):], true
   124  	}
   125  	ctx.ReadDir = fsReadDir(fs)
   126  	ctx.OpenFile = func(path string) (io.ReadCloser, error) {
   127  		return fs.Open(path)
   128  	}
   129  	return ctx
   130  }

   134  func fsReadDir(fs hackpadfs.FS) func(dir string) ([]hackpadfs.FileInfo, error) {
   135  	return func(dir string) ([]hackpadfs.FileInfo, error) {
   136  		dirEntries, err := hackpadfs.ReadDir(fs, dir)
   137  		if err != nil {
   138  			return nil, err
   139  		}
   140  		var infos []hackpadfs.FileInfo
   141  		for _, dirEntry := range dirEntries {
   142  			info, err := dirEntry.Info()
   143  			if err != nil {
   144  				return nil, err
   145  			}
   146  			infos = append(infos, info)
   147  		}
   148  		return infos, nil
   149  	}
   150  }

   152  func findModule(fs hackpadfs.FS, dir string) (moduleName, moduleDir string, err error) {
   153  	for ; dir != "."; dir = path.Dir(dir) {

   164  func getModule(fs hackpadfs.FS, dir string) (moduleName, moduleDir string, ok bool, err error) {
   165  	file := path.Join(dir, "go.mod")
   166  	_, err = hackpadfs.Stat(fs, file)
   167  	if err != nil {
   168  		if errors.Is(err, hackpadfs.ErrNotExist) {
   169  			err = nil
   170  		}
   171  		return "", "", false, err
   172  	}
   173  
   174  	contents, err := hackpadfs.ReadFile(fs, file)
   175  	if err != nil {
   176  		return "", "", false, err
   177  	}
   178  	modFile, err := modfile.Parse(file, contents, nil)
   179  	if err != nil {
   180  		return "", "", false, err
   181  	}
   182  	return modFile.Module.Mod.Path, dir, true, nil
   183  }
