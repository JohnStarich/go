    19  // OSFSWithTemp returns an os.FS instance with 1) the current module's directory and 2) a temporary directory mounted inside.
    20  // Returns the FS paths to both mounts.
    21  func OSFSWithTemp(t *testing.T, relPathToModuleDir string) (_ hackpadfs.FS, workingDirectory, tempDirectory string) {
    22  	t.Helper()
    23  
    24  	wd, err := goos.Getwd()
    25  	require.NoError(t, err)
    26  	wdPath := path.Join(fspath.ToFSPath(wd), relPathToModuleDir)
    27  	osFS, err := os.NewFS().Sub(wdPath)
    28  	require.NoError(t, err)
    29  
    30  	memFS, err := mem.NewFS()
    31  	require.NoError(t, err)
    32  	fs, err := mount.NewFS(memFS)
    33  	require.NoError(t, err)
    34  
    35  	workingDirectory = "work"
    36  	require.NoError(t, hackpadfs.MkdirAll(fs, workingDirectory, testDirPermission))
    37  	require.NoError(t, fs.AddMount(workingDirectory, osFS))
    38  
    39  	tempDirectory = "tmp"
    40  	require.NoError(t, hackpadfs.MkdirAll(fs, tempDirectory, testDirPermission))
    41  	return fs, workingDirectory, tempDirectory
