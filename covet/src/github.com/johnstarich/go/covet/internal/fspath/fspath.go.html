    33  // Similar to filepath.Rel without including OS-dependent behavior.
    34  func Rel(basePath, targetPath string) (string, error) {
    35  	basePath = path.Clean(basePath)
    36  	targetPath = path.Clean(targetPath)
    37  
    38  	common := CommonBase(basePath, targetPath)
    39  	if common == "" {
    40  		return "", errors.New("could not make relative path between basePath and targetPath")
    41  	}
    42  	base := basePath
    43  	base = strings.TrimPrefix(base, common)
    44  	base = strings.TrimPrefix(base, separator)
    45  	target := targetPath
    46  	target = strings.TrimPrefix(target, common)
    47  	target = strings.TrimPrefix(target, separator)
    48  
    49  	switch {
    50  	case base == "" && target == "":
    51  		return ".", nil
    52  	case base == "":
    53  		return target, nil
    54  	default:
    55  		p := strings.Repeat("../", strings.Count(base, separator)+1)
    56  		return path.Join(p, target), nil
    57  	}
