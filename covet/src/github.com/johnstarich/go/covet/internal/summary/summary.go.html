    19  
    20  	var sb strings.Builder
    21  	sb.WriteString("Diff coverage is below target. Add tests for these files:\n")
    22  	tbl := table.NewWriter()
    23  	const coverageColumnIndex = 2
    24  	tbl.SetColumnConfigs([]table.ColumnConfig{
    25  		{Number: coverageColumnIndex, Align: text.AlignCenter},

    27  	tbl.SuppressEmptyColumns()

    30  		"",
    31  		format.Colorize(bold, "Lines"),
    32  		format.Colorize(bold, "Coverage"),
    33  		format.Colorize(bold, "File"),

    35  	for _, f := range uncoveredFiles {
    36  		percent := FileCoverage(f)
    37  		status := coverstatus.New(percent)
    38  		tbl.AppendRow(table.Row{
    39  			format.StatusIcon(status),
    40  			format.ColorizeStatus(status, format.Monospace(formatFraction(f.Covered, f.Uncovered+f.Covered))),
    41  			format.ColorizeStatus(status, format.Monospace(FormatPercent(percent)+" "+formatGraph(percent, format))),
    42  			f.Name,
    43  		})
    44  	}
    45  	sb.WriteString(format.FormatTable(tbl))
    46  	sb.WriteRune('\n')
    47  	return sb.String()
