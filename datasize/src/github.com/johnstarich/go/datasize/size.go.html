   136  	nFloat := new(big.Float).SetInt(n)
   137  	scalarFloat := big.NewFloat(scalar)
   138  	mult := new(big.Float).Mul(nFloat, scalarFloat)

   241  	case s.value.CmpAbs(gigabytes) != -1:
