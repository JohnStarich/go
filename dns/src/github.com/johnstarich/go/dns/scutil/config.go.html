    47  func readMacOSDNS(ctx context.Context, getSCUtilDNS scutilExecutor) (Config, error) {
    48  	output, err := getSCUtilDNS(ctx)
    49  	if err != nil {
    50  		return Config{}, err
    51  	}
    52  
    53  	scanner := bufio.NewScanner(bytes.NewReader(output))
    54  	var config Config
    55  	for scanner.Scan() {
    56  		key, value := splitKeyValue(scanner.Text())
    57  		if strings.Contains(key, "resolver") {
    58  			config.Resolvers = append(config.Resolvers, Resolver{})
    59  			continue
    60  		}
    61  		if len(config.Resolvers) == 0 {
    62  			continue
    63  		}
    64  
    65  		currentResolver := &config.Resolvers[len(config.Resolvers)-1]
    66  		parseAndApplyResolverKey(currentResolver, key, value)
    67  	}
    68  
    69  	return config, nil
    70  }

    72  func parseAndApplyResolverKey(r *Resolver, key, value string) {
    73  	const (
    74  		decimalBase = 10
    75  		maxIntBits  = 64
    76  	)
    77  	switch {
    78  	case strings.Contains(key, "search domain"):
    79  		r.SearchDomain = append(r.SearchDomain, value)
    80  	case strings.Contains(key, "domain"):
    81  		r.Domain = value
    82  	case strings.Contains(key, "flags"):
    83  		for _, flag := range strings.Split(value, ",") {
    84  			r.Flags = append(r.Flags, Flag(strings.TrimSpace(flag)))
    85  		}
    86  	case strings.Contains(key, "if_index"):
    87  		const ifIndexTokenCount = 2
    88  		tokens := strings.SplitN(value, " ", ifIndexTokenCount)
    89  		if len(tokens) == ifIndexTokenCount {
    90  			r.InterfaceName = strings.Trim(tokens[1], "()")
    91  		}
    92  
    93  		i, err := strconv.ParseInt(tokens[0], decimalBase, maxIntBits)
    94  		if err == nil {
    95  			r.InterfaceIndex = int(i)
    96  		}
    97  	case strings.Contains(key, "options"):
    98  		r.MulticastDNS = strings.Contains(value, "mdns")
    99  	case strings.Contains(key, "port"):
   100  		i, err := strconv.ParseInt(value, decimalBase, maxIntBits)
   101  		if err == nil {
   102  			r.Port = int(i)
   103  		}
   104  	case strings.Contains(key, "nameserver"):
   105  		r.Nameservers = append(r.Nameservers, value)
   106  	case strings.Contains(key, "order"):
   107  		i, err := strconv.ParseInt(value, decimalBase, maxIntBits)
   108  		if err == nil {
   109  			r.Order = int(i)
   110  		}
   111  	case strings.Contains(key, "reach"):
   112  		const reachTokenCount = 2
   113  		tokens := strings.SplitN(value, " ", reachTokenCount)
   114  		if len(tokens) == reachTokenCount {
   115  			reach := strings.Trim(tokens[1], "()")
   116  			for _, statusStr := range strings.Split(reach, ",") {
   117  				status := Reach(statusStr)
   118  				if status == Reachable {
   119  					r.reachable = true
   120  				}
   121  				r.Reach = append(r.Reach, status)
   122  			}
   123  		}
   124  	case strings.Contains(key, "timeout"):
   125  		i, err := strconv.ParseInt(value, decimalBase, maxIntBits)
   126  		if err == nil {
   127  			r.Timeout = time.Duration(i) * time.Second
   128  		}
   129  	}
   130  }

   132  func splitKeyValue(line string) (key, value string) {
   133  	const keyValueSplits = 2
   134  	tokens := strings.SplitN(line, ":", keyValueSplits)
   135  	switch len(tokens) {
   136  	case 1:
   137  		return strings.TrimSpace(tokens[0]), ""
   138  	default:
   139  		return strings.TrimSpace(tokens[0]), strings.TrimSpace(tokens[1])
   140  	}
   141  }
