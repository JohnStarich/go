    76  	t.Parallel()
    77  	someConfig := scutil.Config{
    78  		Resolvers: []scutil.Resolver{
    79  			{Nameservers: []string{
    80  				"1.2.3.4",                      // IPv4
    81  				"fe80::8c86:1eff:fe8b:cc64%5d", // IPv6 with zone ID suffix
    82  			}},
    83  		},
    84  	}
    85  	someError := errors.New("some error")
    86  	dialer := newMacOSDialer(Config{})
    87  	callCount := 0
    88  	dialer.readResolvers = func(ctx context.Context) (scutil.Config, error) {
    89  		callCount++
    90  		assert.Less(t, callCount, 2, "Read should not be called more than once")
    91  		return someConfig, someError
    92  	}
    93  	expectedNameservers := []string{
    94  		"1.2.3.4:53",
    95  		"[fe80::8c86:1eff:fe8b:cc64%5d]:53",
    96  	}
    97  	nameservers, err := dialer.ensureNameservers()
    98  	assert.Equal(t, someError, err)
    99  	assert.Equal(t, expectedNameservers, nameservers)
   100  	require.Equal(t, expectedNameservers, dialer.nameservers)
   101  
   102  	nameservers, err = dialer.ensureNameservers() // should not call read again
   103  	assert.Equal(t, expectedNameservers, nameservers)
   104  	assert.NoError(t, err)
   105  }

   111  func TestDNSLookupHost(t *testing.T) {
   112  	t.Parallel()
   113  	const (
   114  		working = true
   115  		failing = false
   116  	)
   117  
   118  	for _, tc := range []struct {
   119  		description string
   120  		nameservers []bool // true is working, false is failing as defined above
   121  		expectErr   string
   122  	}{
   123  		{
   124  			description: "1 working nameserver",
   125  			nameservers: []bool{working},
   126  		},
   127  		{
   128  			description: "1 failing nameserver",
   129  			nameservers: []bool{failing},
   130  			expectErr:   "i/o timeout",
   131  		},
   132  		{
   133  			description: "2 working nameservers",
   134  			nameservers: []bool{working, working},
   135  		},
   136  		{
   137  			description: "2 failing nameservers",
   138  			nameservers: []bool{failing, failing},
   139  			expectErr:   "i/o timeout",
   140  		},
   141  		{
   142  			description: "1 failing, 1 working nameserver",
   143  			nameservers: []bool{failing, working},
   144  		},
   145  		{
   146  			description: "1 working, 1 failing nameserver",
   147  			nameservers: []bool{working, failing},
   148  		},
   149  		{
   150  			description: "many failing, 1 working nameserver",
   151  			nameservers: []bool{
   152  				failing,
   153  				failing,
   154  				failing,
   155  				failing,
   156  				failing,
   157  				failing,
   158  				failing,
   159  				working,
   160  			},
   161  		},
   162  	} {
   163  		tc := tc
   164  		t.Run(tc.description, func(t *testing.T) {
   165  			t.Parallel()
   166  			workingDNS := testhelpers.StartDNSServer(t, testhelpers.DNSConfig{
   167  				ResponseDelay: 1 * time.Second,
   168  				Hostnames: map[string][]string{
   169  					"hi.local.": {"5.6.7.8"},
   170  				},
   171  			})
   172  			failingDNS := "1.2.3.4:53"
   173  
   174  			dialer := testDialer(t)
   175  			for _, workingNS := range tc.nameservers {
   176  				nameserver := workingDNS
   177  				if !workingNS {
   178  					nameserver = failingDNS
   179  				}
   180  				dialer.nameservers = append(dialer.nameservers, nameserver)
   181  			}
   182  
   183  			ctx, cancel := context.WithTimeout(context.Background(), testTimeout)
   184  			defer cancel()
   185  
   186  			conn, err := dialer.DialContext(ctx, "ignored", "ignored")
   187  			require.NoError(t, err)
   188  			assert.IsType(t, (*staggercast.Conn)(nil), conn)
   189  			conn.Close()
   190  
   191  			res := &net.Resolver{PreferGo: true, Dial: dialer.DialContext}
   192  
   193  			addrs, err := res.LookupHost(ctx, "hi.local")
   194  			if tc.expectErr != "" {
   195  				require.Error(t, err)
   196  				assert.Contains(t, err.Error(), tc.expectErr)
   197  				return
   198  			}
   199  			assert.NoError(t, err)
   200  			assert.Equal(t, []string{"5.6.7.8"}, addrs)
   201  		})
   202  	}
   203  }

   233  func TestReorderNameservers(t *testing.T) {
   234  	t.Parallel()
   235  	addr := testhelpers.StartDNSServer(t, testhelpers.DNSConfig{
   236  		Hostnames: map[string][]string{
   237  			"hi.local.": {"5.6.7.8"},
   238  		},
   239  	})
   240  
   241  	const initialDelay = 3 * time.Second
   242  	dialer := newMacOSDialer(Config{
   243  		InitialNameserverDelay: initialDelay,
   244  		Logger:                 newTestLogger(t),
   245  	})
   246  	dialer.nameservers = []string{"1.2.3.4:53", addr}
   247  	res := &net.Resolver{PreferGo: true, Dial: dialer.DialContext}
   248  	ctx, cancel := context.WithTimeout(context.Background(), testTimeout)
   249  	start := time.Now()
   250  	addrs, err := res.LookupHost(ctx, "hi.local")
   251  	duration := time.Since(start)
   252  	cancel() // explicitly cancel to reorder nameservers now
   253  
   254  	require.NoError(t, err)
   255  	assert.Equal(t, []string{"5.6.7.8"}, addrs)
   256  	assert.GreaterOrEqual(t, int(duration), int(initialDelay), "First request should take longer than initial delay")
   257  	time.Sleep(time.Second)
   258  	assert.Equal(t, []string{addr, "1.2.3.4:53"}, dialer.nameservers)
   259  
   260  	ctx, cancel = context.WithTimeout(context.Background(), testTimeout)
   261  	defer cancel()
   262  	start = time.Now()
   263  	addrs, err = res.LookupHost(ctx, "hi.local")
   264  	duration = time.Since(start)
   265  
   266  	require.NoError(t, err)
   267  	assert.Equal(t, []string{"5.6.7.8"}, addrs)
   268  	assert.LessOrEqual(t, int(duration), int(initialDelay/100), "Second request should reorder and complete almost instantly")
   269  }
