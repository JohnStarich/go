    52  // New combines multiple packet connections (net.Conn or net.PacketConn) and returns a single, staggerable connection
    53  func New(conns []PacketConn) *Conn {
    54  	if len(conns) == 0 {
    55  		panic("connection count must be non-zero")
    56  	}
    57  	replay := make([]chan struct{}, len(conns))
    58  	for i := range replay {
    59  		replay[i] = make(chan struct{})
    60  		close(replay[i])
    61  	}
    62  	return &Conn{
    63  		conns: conns,
    64  
    65  		// by default, fire all connections at once
    66  		connCount:    atomic.NewUint64(uint64(len(conns))),
    67  		replay:       replay,
    68  		tickerCancel: func() {},
    69  
    70  		firstResponder: atomic.NewInt64(-1), // -1 marks no known first responder, so connection #0 can be designated first
    71  	}
    72  }

    74  // Stagger controls when secondary connections are attempted after the first. For any new enabled connection, the last Write and SetDeadlines are replayed onto it.
    75  // If Stagger is called after the connection is started, then the behavior is undefined.
    76  // First connection is attempted immediately, then all future connections will be enabled when receiving from ticker.
    77  //
    78  // Example: Create and wrap a 'time.Ticker.C' with a duration of 1 second.
    79  // The first connection is attempted immediately. For every 1 second the Read hasn't returned, an additional connection is attempted.
    80  // Once a connection succeeds, the result is returned immediately from Read.
    81  func (s *Conn) Stagger(ticker <-chan struct{}, cancel context.CancelFunc) {
    82  	s.replayMu.Lock()
    83  
    84  	ctx, tickerCancel := context.WithCancel(context.Background())
    85  	s.tickerCancel = tickerCancel
    86  
    87  	totalLength := uint64(len(s.conns))
    88  	s.replay = make([]chan struct{}, totalLength)
    89  	for i := range s.replay {
    90  		s.replay[i] = make(chan struct{})
    91  	}
    92  	close(s.replay[0]) // initial connection should immediately fire
    93  	connCount := atomic.NewUint64(1)
    94  	s.connCount = connCount
    95  
    96  	// listen for ticks and fire off replays on new connections
    97  	go func() {
    98  		// only reference params and local vars here, no state from 'p'
    99  		for {
   100  			select {
   101  			case <-ticker:
   102  				count := connCount.Inc()
   103  				if count > totalLength {
   104  					// finished enabling all connections
   105  					cancel()
   106  					return
   107  				}
   108  				go s.runReplay(count - 1)
   109  			case <-ctx.Done():
   110  				cancel()
   111  				return
   112  			}
   113  		}
   114  	}()
   115  	s.replayMu.Unlock()
   116  }

   142  // runReplay synchronously reapplies the last Deadlines and the last Write on a best-effort basis
   143  func (s *Conn) runReplay(connIndex uint64) {
   144  	s.replayMu.RLock()
   145  	if !s.lastDeadline.IsZero() {
   146  		_ = s.conns[connIndex].SetDeadline(s.lastDeadline)
   147  	}
   148  	if !s.lastReadDeadline.IsZero() {
   149  		_ = s.conns[connIndex].SetReadDeadline(s.lastReadDeadline)
   150  	}
   151  	if !s.lastWriteDeadline.IsZero() {
   152  		_ = s.conns[connIndex].SetWriteDeadline(s.lastWriteDeadline)
   153  	}
   154  	b := s.lastWrite
   155  	s.replayMu.RUnlock()
   156  	if b != nil {
   157  		_, _ = s.conns[connIndex].Write(b)
   158  	}
   159  	close(s.replay[connIndex]) // fire off any pending iter's
   160  }

   170  func (s *Conn) iter(op connOp, fn func(conn PacketConn) (keepGoing bool, err error)) error {
   171  	ctx, cancel := context.WithCancel(context.Background())
   172  	defer cancel()
   173  	done := make(chan struct{}, len(s.conns))
   174  	errs := make(chan error, len(s.conns))
   175  	for ix, conn := range s.conns {
   176  		go func(ix int, conn PacketConn) {
   177  			select {
   178  			case <-s.replay[ix]:
   179  			case <-ctx.Done():
   180  				return
   181  			}
   182  
   183  			keepGoing, err := fn(conn)
   184  			if err != nil {
   185  				errs <- err
   186  			} else if op == readOp || op == readFromOp {
   187  				s.firstResponder.CompareAndSwap(-1, int64(ix))
   188  			}
   189  			done <- struct{}{}
   190  			if !keepGoing {
   191  				cancel()
   192  			}
   193  		}(ix, conn)
   194  	}
   195  	for i := uint64(0); i < s.getConnCount(); i++ {
   196  		select {
   197  		case <-done:
   198  		case <-ctx.Done():
   199  			goto ctxDone
   200  		}
   201  	}
   202  
   203  ctxDone:
   204  	if uint64(len(errs)) >= s.getConnCount() {
   205  		// only return an error if all conns failed
   206  		err := <-errs
   207  		return errors.Wrapf(err, "all connections have failed for %q", op)
   208  	}
   209  	return nil
   210  }

   212  func (s *Conn) Read(b []byte) (n int, err error) {
   213  	type byteResp struct {
   214  		n int
   215  		b []byte
   216  	}
   217  	success := make(chan byteResp, len(s.conns))
   218  	failure := make(chan byteResp, len(s.conns))
   219  	err = s.iter(readOp, func(conn PacketConn) (bool, error) {
   220  		buf := make([]byte, len(b))
   221  		n, err := conn.Read(buf)
   222  		if err == nil {
   223  			success <- byteResp{n: n, b: buf}
   224  		} else {
   225  			failure <- byteResp{n: n, b: buf}
   226  		}
   227  		return err != nil, err
   228  	})
   229  	select {
   230  	case resp := <-success:
   231  		n = resp.n
   232  		copy(b, resp.b)
   233  	case resp := <-failure:
   234  		n = resp.n
   235  		copy(b, resp.b)
   236  	}
   237  	return
   238  }
