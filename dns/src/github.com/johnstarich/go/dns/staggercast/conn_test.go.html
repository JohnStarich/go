   159  func TestStagger(t *testing.T) {
   160  	t.Parallel()
   161  
   162  	var servers []string
   163  	addr, cancel := testhelpers.StartDNSServer(t, testhelpers.DNSConfig{
   164  		ResponseDelay: 30 * time.Second,
   165  		Hostnames: map[string][]string{
   166  			"hi.local.": {"1.2.3.4"},
   167  		},
   168  	})
   169  	defer cancel()
   170  	servers = append(servers, addr)
   171  	addr, cancel = testhelpers.StartDNSServer(t, testhelpers.DNSConfig{
   172  		Hostnames: map[string][]string{
   173  			"hi.local.": {"5.6.7.8"},
   174  		},
   175  	})
   176  	defer cancel()
   177  	servers = append(servers, addr)
   178  
   179  	t.Run("stagger never enables", func(t *testing.T) {
   180  		res := &net.Resolver{
   181  			PreferGo: true,
   182  			Dial: func(ctx context.Context, network, address string) (net.Conn, error) {
   183  				var conns []PacketConn
   184  				for _, addr := range servers {
   185  					conns = append(conns, dialUDP(t, addr))
   186  				}
   187  				conn := New(conns)
   188  				conn.Stagger(nil, func() {}) // effectively disable all further connections
   189  				return conn, nil
   190  			},
   191  		}
   192  
   193  		ctx, cancel := context.WithTimeout(context.Background(), testTimeout)
   194  		defer cancel()
   195  
   196  		addrs, err := res.LookupHost(ctx, "hi.local")
   197  		assert.Empty(t, addrs)
   198  		require.Error(t, err)
   199  		assert.Contains(t, err.Error(), "all connections have failed for \"write\": write udp")
   200  	})
   201  
   202  	t.Run("stagger eventually succeeds", func(t *testing.T) {
   203  		const delay = 1 * time.Second
   204  		res := &net.Resolver{
   205  			PreferGo: true,
   206  			Dial: func(ctx context.Context, network, address string) (net.Conn, error) {
   207  				var conns []PacketConn
   208  				for _, addr := range servers {
   209  					conns = append(conns, dialUDP(t, addr))
   210  				}
   211  				conn := New(conns)
   212  
   213  				// stagger new connections with a ticker of 'delay' interval
   214  				ctx, cancel := context.WithCancel(ctx)
   215  				c := make(chan struct{})
   216  				go func() {
   217  					ticker := time.NewTicker(delay)
   218  					for {
   219  						select {
   220  						case <-ticker.C:
   221  							c <- struct{}{}
   222  						case <-ctx.Done():
   223  							ticker.Stop()
   224  							return
   225  						}
   226  					}
   227  				}()
   228  				conn.Stagger(c, cancel) // effectively disable all further connections
   229  				return conn, nil
   230  			},
   231  		}
   232  
   233  		ctx, cancel := context.WithTimeout(context.Background(), testTimeout)
   234  		defer cancel()
   235  
   236  		start := time.Now()
   237  		addrs, err := res.LookupHost(ctx, "hi.local")
   238  		d := time.Since(start)
   239  		assert.Equal(t, []string{"5.6.7.8"}, addrs)
   240  		assert.NoError(t, err)
   241  		assert.LessOrEqual(t, int(delay), int(d), "DNS should not resolve before second connection is enabled")
   242  		assert.GreaterOrEqual(t, int(testTimeout), int(d), "DNS should resolve before the test times out")
   243  	})
