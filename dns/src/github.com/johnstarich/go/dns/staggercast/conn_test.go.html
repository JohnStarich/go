   158  func TestStagger(t *testing.T) {
   159  	t.Parallel()
   160  
   161  	startServers := func(t *testing.T) []string {
   162  		const slowServerDelay = 30 * time.Second
   163  		var servers []string
   164  		addr := testhelpers.StartDNSServer(t, testhelpers.DNSConfig{
   165  			ResponseDelay: slowServerDelay,
   166  			Hostnames: map[string][]string{
   167  				"hi.local.": {"1.2.3.4"},
   168  			},
   169  		})
   170  		servers = append(servers, addr)
   171  		addr = testhelpers.StartDNSServer(t, testhelpers.DNSConfig{
   172  			Hostnames: map[string][]string{
   173  				"hi.local.": {"5.6.7.8"},
   174  			},
   175  		})
   176  		servers = append(servers, addr)
   177  		return servers
   178  	}
   179  
   180  	t.Run("stagger never enables", func(t *testing.T) {
   181  		t.Parallel()
   182  		servers := startServers(t)
   183  		res := &net.Resolver{
   184  			PreferGo: true,
   185  			Dial: func(ctx context.Context, network, address string) (net.Conn, error) {
   186  				var conns []PacketConn
   187  				for _, addr := range servers {
   188  					conns = append(conns, dialUDP(t, addr))
   189  				}
   190  				conn := New(conns)
   191  				conn.Stagger(nil, func() {}) // effectively disable all further connections
   192  				return conn, nil
   193  			},
   194  		}
   195  
   196  		ctx, cancel := context.WithTimeout(context.Background(), testTimeout)
   197  		defer cancel()
   198  
   199  		addrs, err := res.LookupHost(ctx, "hi.local")
   200  		assert.Empty(t, addrs)
   201  		require.Error(t, err)
   202  		assert.Contains(t, err.Error(), "all connections have failed for \"write\": write udp")
   203  	})
   204  
   205  	t.Run("stagger eventually succeeds", func(t *testing.T) {
   206  		t.Parallel()
   207  		if os.Getenv("CI") == "true" && runtime.GOOS == "darwin" {
   208  			t.Skip("DNS timeouts on macOS in CI don't work very well.") // FIXME: This doesn't seem to work on macOS in CI, despite working fine locally.
   209  		}
   210  
   211  		servers := startServers(t)
   212  		const delay = 1 * time.Second
   213  		res := &net.Resolver{
   214  			PreferGo: true,
   215  			Dial: func(ctx context.Context, network, address string) (net.Conn, error) {
   216  				var conns []PacketConn
   217  				for _, addr := range servers {
   218  					conns = append(conns, dialUDP(t, addr))
   219  				}
   220  				conn := New(conns)
   221  
   222  				// stagger new connections with a ticker of 'delay' interval
   223  				ctx, cancel := context.WithCancel(ctx)
   224  				c := make(chan struct{})
   225  				go func() {
   226  					ticker := time.NewTicker(delay)
   227  					for {
   228  						select {
   229  						case <-ticker.C:
   230  							c <- struct{}{}
   231  						case <-ctx.Done():
   232  							ticker.Stop()
   233  							return
   234  						}
   235  					}
   236  				}()
   237  				conn.Stagger(c, cancel) // effectively disable all further connections
   238  				return conn, nil
   239  			},
   240  		}
   241  
   242  		ctx, cancel := context.WithTimeout(context.Background(), testTimeout)
   243  		defer cancel()
   244  
   245  		start := time.Now()
   246  		addrs, err := res.LookupHost(ctx, "hi.local")
   247  		d := time.Since(start)
   248  		assert.NoError(t, err)
   249  		assert.Equal(t, []string{"5.6.7.8"}, addrs)
   250  		assert.LessOrEqual(t, int(delay), int(d), "DNS should not resolve before second connection is enabled")
   251  		assert.GreaterOrEqual(t, int(testTimeout), int(d), "DNS should resolve before the test times out")
   252  	})
   253  
   254  	t.Run("stagger enables all (almost) instantly", func(t *testing.T) {
   255  		t.Parallel()
   256  		servers := startServers(t)
   257  		res := &net.Resolver{
   258  			PreferGo: true,
   259  			Dial: func(ctx context.Context, network, address string) (net.Conn, error) {
   260  				var conns []PacketConn
   261  				for _, addr := range servers {
   262  					conns = append(conns, dialUDP(t, addr))
   263  				}
   264  				conn := New(conns)
   265  
   266  				// stagger new connections with a ticker of 'delay' interval
   267  				c := make(chan struct{})
   268  				close(c)
   269  				conn.Stagger(c, func() {}) // effectively disable all further connections
   270  				return conn, nil
   271  			},
   272  		}
   273  
   274  		ctx, cancel := context.WithTimeout(context.Background(), testTimeout)
   275  		defer cancel()
   276  
   277  		start := time.Now()
   278  		addrs, err := res.LookupHost(ctx, "hi.local")
   279  		d := time.Since(start)
   280  		assert.NoError(t, err)
   281  		assert.Equal(t, []string{"5.6.7.8"}, addrs)
   282  		assert.Less(t, int(d), int(float64(testTimeout)*0.01), "Second connection should be used almost immediately")
   283  	})
