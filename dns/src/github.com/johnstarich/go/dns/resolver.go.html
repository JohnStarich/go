    84  func (m *macOSDialer) ensureNameservers() ([]string, error) {
    85  	m.nameserversMu.RLock()
    86  	nameservers := m.nameservers
    87  	m.nameserversMu.RUnlock()
    88  	if len(nameservers) != 0 {
    89  		return nameservers, nil
    90  	}
    91  	m.nameserversMu.Lock()
    92  	defer m.nameserversMu.Unlock()
    93  	if len(m.nameservers) != 0 { // check again, could change while waiting on lock
    94  		return m.nameservers, nil
    95  	}
    96  
    97  	const maxTimeout = 5 * time.Second
    98  	ctx, cancel := context.WithTimeout(context.Background(), maxTimeout)
    99  	defer cancel()
   100  	m.Logger.Info("Reading macOS DNS config from 'scutil'...")
   101  	cfg, err := m.readResolvers(ctx)
   102  	for _, resolver := range cfg.Resolvers {
   103  		port := resolver.Port
   104  		if port == 0 {
   105  			port = 53
   106  		}
   107  		for _, nameserver := range resolver.Nameservers {
   108  			m.nameservers = append(m.nameservers, dialableNameserver(nameserver)+":"+fmt.Sprint(port))
   109  		}
   110  	}
   111  	m.Logger.Info("Finished reading macOS DNS config from 'scutil'", zap.Error(err))
   112  	return m.nameservers, err
   113  }

   115  func (m *macOSDialer) DialContext(ctx context.Context, network, address string) (net.Conn, error) {
   116  	nameservers, err := m.ensureNameservers()
   117  	if err != nil {
   118  		m.Logger.Error("Failed looking up macOS resolvers, falling back to builtin DNS dialer", zap.Error(err))
   119  		return m.dialer.DialContext(ctx, network, address)
   120  	}
   121  
   122  	conn, err := m.dialAll(ctx, nameservers)
   123  	if err != nil {
   124  		m.Logger.Error("Failed dialing macOS nameservers, falling back to builtin DNS dialer", zap.Error(err))
   125  		return m.dialer.DialContext(ctx, network, address)
   126  	}
   127  
   128  	return conn, nil
   129  }

   131  func (m *macOSDialer) dialAll(ctx context.Context, nameservers []string) (net.Conn, error) {
   132  	type dialResp struct {
   133  		ix   int
   134  		conn net.Conn
   135  	}
   136  	conns := make(chan dialResp, len(nameservers))
   137  
   138  	var wait sync.WaitGroup
   139  	wait.Add(len(nameservers))
   140  
   141  	m.Logger.Debug("Dialing all nameservers")
   142  	for ix, nameserver := range nameservers {
   143  		go func(ix int, nameserver string) {
   144  			conn, err := m.dialer.DialContext(ctx, "udp", nameserver)
   145  			if err != nil {
   146  				m.Logger.Warn("Error dialing nameserver", zap.String("nameserver", nameserver), zap.Int("index", ix), zap.Error(err))
   147  			} else {
   148  				conns <- dialResp{ix: ix, conn: conn}
   149  			}
   150  			wait.Done()
   151  		}(ix, nameserver)
   152  	}
   153  	wait.Wait()
   154  	select {
   155  	case <-ctx.Done():
   156  		return nil, ctx.Err()
   157  	default:
   158  	}
   159  	close(conns)
   160  
   161  	var allSuccessfulResps []dialResp
   162  	for resp := range conns {
   163  		allSuccessfulResps = append(allSuccessfulResps, resp)
   164  	}
   165  	if len(allSuccessfulResps) == 0 {
   166  		return nil, errors.New("error dialing all nameservers")
   167  	}
   168  	sort.Slice(allSuccessfulResps, func(a, b int) bool { // preserve original nameserver order
   169  		return allSuccessfulResps[a].ix < allSuccessfulResps[b].ix
   170  	})
   171  
   172  	var allSuccessfulConns []staggercast.PacketConn
   173  	for _, resp := range allSuccessfulResps {
   174  		allSuccessfulConns = append(allSuccessfulConns, resp.conn.(staggercast.PacketConn)) // UDP connections must also support net.PacketConn
   175  	}
   176  
   177  	staggerConn := staggercast.New(allSuccessfulConns)
   178  	staggerConn.Stagger(staggerTicker(m.InitialNameserverDelay, m.NextNameserverInterval, m.Logger))
   179  	go m.reorderNameservers(ctx, staggerConn)
   180  	return staggerConn, nil
   181  }

   183  func staggerTicker(initialDelay, d time.Duration, logger *zap.Logger) (<-chan struct{}, context.CancelFunc) {
   184  	ctx, cancel := context.WithCancel(context.Background())
   185  	c := make(chan struct{})
   186  	go func() {
   187  		index := 1
   188  		timer := time.NewTimer(initialDelay)
   189  		select {
   190  		case <-timer.C:
   191  			logger.Debug("scatter: Enabling connection", zap.Int("index", index))
   192  			index++
   193  			c <- struct{}{}
   194  		case <-ctx.Done():
   195  			timer.Stop()
   196  			return
   197  		}
   198  		timer.Stop()
   199  
   200  		ticker := time.NewTicker(d)
   201  		for {
   202  			select {
   203  			case <-ctx.Done():
   204  				ticker.Stop()
   205  				close(c)
   206  				return
   207  			case <-ticker.C:
   208  				logger.Debug("scatter: Enabling connection", zap.Int("index", index))
   209  				index++
   210  				c <- struct{}{}
   211  			}
   212  		}
   213  	}()
   214  	return c, cancel
   215  }

   217  func (m *macOSDialer) reorderNameservers(ctx context.Context, conn *staggercast.Conn) {
   218  	if _, ok := ctx.Deadline(); !ok {
   219  		m.Logger.Debug("Skipping nameserver reorder, no deadline on context")
   220  		return
   221  	}
   222  
   223  	<-ctx.Done()
   224  	stats := conn.Stats()
   225  	if stats.FastestRemoteIndex == 0 {
   226  		return
   227  	}
   228  
   229  	m.nameserversMu.Lock()
   230  	defer m.nameserversMu.Unlock()
   231  
   232  	fastestRemoteAddr := stats.FastestRemote.String()
   233  	fastIndexNS := m.nameservers[stats.FastestRemoteIndex]
   234  	if fastIndexNS != fastestRemoteAddr {
   235  		m.Logger.Debug("Fastest remote already reordered", zap.String("remote", stats.FastestRemote.String()), zap.String("nsIndex", fastIndexNS))
   236  		return
   237  	}
   238  	newNS := []string{fastIndexNS}
   239  	newNS = append(newNS, m.nameservers[:stats.FastestRemoteIndex]...)
   240  	newNS = append(newNS, m.nameservers[stats.FastestRemoteIndex+1:]...)
   241  	m.nameservers = newNS
   242  	m.Logger.Debug("Reordering fastest nameserver to the front", zap.Strings("nameservers", newNS))
   243  }

   245  func dialableNameserver(nameserver string) string {
   246  	ipStr := nameserver
   247  	if i := strings.IndexRune(nameserver, '%'); i != -1 {
   248  		// remove optional IPv6 zone ID suffix to parse IP
   249  		ipStr = nameserver[:i]
   250  	}
   251  	isIPv4 := net.ParseIP(ipStr).To4() != nil
   252  	if !isIPv4 {
   253  		return fmt.Sprintf("[%s]", nameserver)
   254  	}
   255  	return nameserver
   256  }
