    25  	}
    26  	if i := strings.IndexRune(pkg.Path, '@'); i != -1 {
    27  		pkg.Path, pkg.ModuleVersion = pkg.Path[:i], pkg.Path[i+1:]
    28  	}
    29  	if strings.HasSuffix(pkg.Path, "/...") {
    30  		return pkg, fmt.Errorf("package pattern must not use the '/...' operator: %q", pkg.Path)

    32  	if runtime.GOOS != goosWindows && strings.HasPrefix(pkg.Path, a.staticOSHomeDir+"/") { // Tilde '~' expansion is not supported on Windows yet.
    33  		// Canonicalize home directory to '~' for better cross-machine bin support, as home directories can change from user to user.
    34  		pkg.Path = homeDir + strings.TrimPrefix(pkg.Path, a.staticOSHomeDir)
    35  	}
    36

    38  	if i := strings.LastIndexAny(pkg.Path, `\/`); i != -1 {
