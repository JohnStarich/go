   268  func fromEnv(env []string) map[string]string {
   269  	m := make(map[string]string)
   270  	for _, envPair := range env {
   271  		equalIndex := strings.IndexRune(envPair, '=')
   272  		if equalIndex == -1 {
   273  			panic("Invalid env key-value pair: " + envPair)
   274  		}
   275  		key, value := envPair[:equalIndex], envPair[equalIndex+1:]
   276  		m[key] = value
   277  	}
   278  	return m
   279  }

   301  // ToOSPath implements hackpadfs.MountFS
   302  func (fs *fsWithOSPath) ToOSPath(name string) (string, error) { //nolint:unparam // Implements interface, cannot remove param.
   303  	for osPath, fsPath := range fs.osToFSPaths {
   304  		if name == fsPath {
   305  			return osPath, nil
   306  		}
   307  	}
   308  	return name, nil

   311  // FromOSPath implements hackpadfs.MountFS
   312  func (fs *fsWithOSPath) FromOSPath(name string) (string, error) { //nolint:unparam // Implements interface, cannot remove param.
   313  	fsPath, ok := fs.osToFSPaths[name]
   314  	if !ok {
   315  		return name, nil
   316  	}
   317  	return fsPath, nil
   318  }
