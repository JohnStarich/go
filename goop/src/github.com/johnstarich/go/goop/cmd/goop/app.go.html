    40  func newApp(outWriter, errWriter io.Writer) (App, error) {
    41  	configDir, err := os.UserConfigDir()
    42  	if err != nil {
    43  		return App{}, err
    44  	}
    45  	cacheDir, err := os.UserCacheDir()
    46  	if err != nil {
    47  		return App{}, err
    48  	}
    49  	osHomeDir, err := os.UserHomeDir()
    50  	if err != nil {
    51  		return App{}, err
    52  	}
    53  
    54  	// convert OS paths to FS paths
    55  	osPaths := []*string{
    56  		&configDir,
    57  		&cacheDir,
    58  	}
    59  	fs := osfs.NewFS()
    60  	for _, p := range osPaths {
    61  		*p, err = fs.FromOSPath(*p)
    62  		if err != nil {
    63  			return App{}, err
    64  		}
    65  	}
    66  
    67  	const configBin = "bin"
    68  	return App{
    69  		errWriter:       errWriter,
    70  		fs:              fs,
    71  		getEnv:          os.Getenv,
    72  		lookPath:        exec.LookPath,
    73  		outWriter:       outWriter,
    74  		runCmd:          runCmd,
    75  		staticBinDir:    path.Join(configDir, appName, configBin),
    76  		staticCacheDir:  path.Join(cacheDir, appName),
    77  		staticOSHomeDir: osHomeDir,
