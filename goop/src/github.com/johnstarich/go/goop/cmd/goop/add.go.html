    82  	scriptOSPath, err := a.toOSPath(scriptPath)

    85  	}

    88  		fmt.Fprintf(a.errWriter, "WARNING: Failed to find %q on PATH. Check to ensure the directory %q is added to your PATH environment variable.\n", name, path.Dir(scriptOSPath))
    89  	}
    90  	if scriptOSPath != executableOSPath {
    91  		fmt.Fprintf(a.errWriter, "WARNING: Executable on PATH for name %q does not match install location: %q != %q\n", name, executableOSPath, scriptOSPath)
    92  	}
