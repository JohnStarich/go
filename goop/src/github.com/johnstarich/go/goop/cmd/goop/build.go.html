    23  	desiredPath := path.Join(a.packageInstallDir(name), name) + systemExt(goos)
    24  	info, err := hackpadfs.Stat(a.fs, desiredPath)
    25  	if err != nil && !errors.Is(err, hackpadfs.ErrNotExist) {
    26  		return "", err
    27  	}
    28  	if err == nil && info.Mode().IsRegular() && !alwaysBuild {
    29  		shouldRebuild, err := a.shouldRebuild(info, pkg)
    30  		if err != nil {
    31  			return "", err
    32  		}
    33  		if !shouldRebuild {
    34  			return desiredPath, nil
    35  		}
    36  	}
    37  
    38  	fmt.Fprintf(a.errWriter, "Building %q...\n", pkg.Path)

   114  func findBinary(fs hackpadfs.FS, installDir string) (string, bool, error) {
   115  	dirEntries, err := hackpadfs.ReadDir(fs, installDir)
   116  	if err != nil && !errors.Is(err, hackpadfs.ErrNotExist) {
   117  		return "", false, err
   118  	}
   119  	for _, entry := range dirEntries {
   120  		if entry.Type().IsRegular() {
   121  			return path.Join(installDir, entry.Name()), true, nil
   122  		}
   123  	}
   124  	return "", false, nil

   127  func (a App) shouldRebuild(binaryInfo hackpadfs.FileInfo, pkg Package) (bool, error) {
   128  	filePath, isFilePath := a.packageFilePath(pkg)
   129  	if !isFilePath {
   130  		// remote module, assume up-to-date
   131  		return false, nil
   132  	}
   133  
   134  	// local module, find out if the build is stale
   135  	binaryModTime := binaryInfo.ModTime()
   136  	fsPath, err := a.fromOSPath(filePath)
   137  	if err != nil {
   138  		return false, err
   139  	}
   140  	moduleRoot, err := moduleRoot(a.fs, fsPath)
   141  	if err != nil {
   142  		return false, err
   143  	}
   144  	return hasNewerModTime(a.fs, moduleRoot, binaryModTime)
   145  }

   147  func moduleRoot(fs hackpadfs.FS, p string) (string, error) {
   148  	const goMod = "go.mod"
   149  	_, err := hackpadfs.Stat(fs, path.Join(p, goMod))
   150  	if err == nil {
   151  		return p, nil
   152  	}
   153  	parentPath := path.Dir(p)
   154  	if parentPath == p {
   155  		return "", errors.Errorf("go.mod not found for package: %q", p)
   156  	}
   157  	root, parentErr := moduleRoot(fs, parentPath)
   158  	if parentErr != nil {
   159  		// if parent also failed to find module, return original error
   160  		return "", err
   161  	}
   162  	return root, nil
   163  }

   165  func hasNewerModTime(fs hackpadfs.FS, root string, baseModTime time.Time) (bool, error) {
   166  	hasNewerModTime := false
   167  	err := hackpadfs.WalkDir(fs, root, func(path string, d hackpadfs.DirEntry, err error) error {
   168  		if hasNewerModTime {
   169  			return hackpadfs.SkipDir
   170  		}
   171  		if err != nil || path == root {
   172  			return err
   173  		}
   174  		info, err := d.Info()
   175  		if err != nil {
   176  			return err
   177  		}
   178  		modTime := info.ModTime()
   179  		if modTime.After(baseModTime) {
   180  			hasNewerModTime = true
   181  			return hackpadfs.SkipDir
   182  		}
   183  		return nil
   184  	})
   185  	return hasNewerModTime, err
   186  }
