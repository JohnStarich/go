    23  	desiredPath := path.Join(a.packageInstallDir(name), name) + systemExt(goos)
    24  	info, err := hackpadfs.Stat(a.fs, desiredPath)
    25  	if err != nil && !errors.Is(err, hackpadfs.ErrNotExist) {
    26  		return "", err
    27  	}
    28  	if err == nil && info.Mode().IsRegular() && !alwaysBuild {
    29  		shouldRebuild, err := a.shouldRebuild(info, pkg)
    30  		if err != nil {

    34  			return desiredPath, nil
    35  		}
    36  	}

   112  func findBinary(fs hackpadfs.FS, installDir string) (string, bool, error) {
   113  	dirEntries, err := hackpadfs.ReadDir(fs, installDir)
   114  	if err != nil && !errors.Is(err, hackpadfs.ErrNotExist) {
   115  		return "", false, err
   116  	}
   117  	for _, entry := range dirEntries {

   125  func (a App) shouldRebuild(binaryInfo hackpadfs.FileInfo, pkg Package) (bool, error) {
   126  	filePath, isFilePath := a.packageFilePath(pkg)
   127  	if !isFilePath {
   128  		// remote module, assume up-to-date
   129  		return false, nil
   130  	}
   131  
   132  	// local module, find out if the build is stale
   133  	binaryModTime := binaryInfo.ModTime()
   134  	fsPath, err := a.fromOSPath(filePath)
   135  	if err != nil {
   136  		return false, err
   137  	}
   138  	moduleRoot, err := moduleRoot(a.fs, fsPath)
   139  	if err != nil {
   140  		return false, err
   141  	}
   142  	return hasNewerModTime(a.fs, moduleRoot, binaryModTime)
   143  }

   145  func moduleRoot(fs hackpadfs.FS, p string) (string, error) {
   146  	const goMod = "go.mod"
   147  	_, err := hackpadfs.Stat(fs, path.Join(p, goMod))
   148  	if err == nil {
   149  		return p, nil
   150  	}
   151  	parentPath := path.Dir(p)
   152  	if parentPath == p {
   153  		return "", errors.Errorf("go.mod not found for package: %q", p)
   154  	}
   155  	root, parentErr := moduleRoot(fs, parentPath)
   156  	if parentErr != nil {
   157  		// if parent also failed to find module, return original error
   158  		return "", err
   159  	}
   160  	return root, nil
   161  }

   163  func hasNewerModTime(fs hackpadfs.FS, root string, baseModTime time.Time) (bool, error) {
   164  	hasNewerModTime := false
   165  	err := hackpadfs.WalkDir(fs, root, func(path string, d hackpadfs.DirEntry, err error) error {
   166  		if hasNewerModTime {
   167  			return hackpadfs.SkipDir
   168  		}
   169  		if err != nil || path == root {
   170  			return err
   171  		}
   172  		info, err := d.Info()
   173  		if err != nil {
   174  			return err
   175  		}
   176  		modTime := info.ModTime()
   177  		if modTime.After(baseModTime) {
   178  			hasNewerModTime = true
   179  			return hackpadfs.SkipDir
   180  		}
   181  		return nil
   182  	})
   183  	return hasNewerModTime, err
   184  }
